006package org.dellroad.hl7;
007
008import java.io.Serializable;
009import java.util.regex.Matcher;
010import java.util.regex.Pattern;
011
012/**
013 * Represents an HL7 message.
014 */
015@SuppressWarnings("serial")
016public class HL7Message implements Serializable {
017
018    /**
019     * The character that separates segments in an HL7 message.
020     */
021    public static final char SEGMENT_TERMINATOR = '\r';
022
023    private static final Pattern FIELD_NAME_PATTERN = Pattern.compile("([\\p{Alnum}]{3})\\.[0-9]+");
024
025    private static final Pattern VALUE_NAME_PATTERN = Pattern.compile("(" + FIELD_NAME_PATTERN + ")(\\.([0-9]+)(\\.([0-9]+))?)?");
026
027    /**
028     * The segments in this message. This list always contains at least
029     * one element, namely the {@link MSHSegment}.
030     */
031    protected final HL7SegmentList segments;
032
033    /**
034     * Construct a new HL7 message containing only the given MSH segment.
035     *
036     * @param msh MSH segment
037     */
038    public HL7Message(MSHSegment msh) {
039        this.segments = new HL7SegmentList(msh);
040    }
041
042    /**
043     * Construct an empty HL7 message.
044     *
045     * <p>
046     * A MSH segment will be automatically added.
047     *
048     * @param seps separator and escape characters to use for this message
049     */
050    public HL7Message(HL7Seps seps) {
051        this.segments = new HL7SegmentList(new MSHSegment(seps));
052    }
053
054    /**
055     * Convenience constructor. Equivalent to:
056     *  <blockquote>
057     *  <code>HL7Message(HL7Seps.DEFAULT)</code>
058     *  <blockquote>
059     */
060    public HL7Message() {
061        this(HL7Seps.DEFAULT);
062    }
063
064    /**
065     * Parsing constructor. Constructs an HL7 message by parsing the given string.
066     * Segments must be separated with a carriage return character.
067     *
068     * @throws HL7ContentException if the string is invalid
069     */
070    public HL7Message(String msg) throws HL7ContentException {
071
072        // Eliminate trailing CR's
073        int len = msg.length();
074        while (len > 0 && msg.charAt(len - 1) == SEGMENT_TERMINATOR)
075            msg = msg.substring(0, --len);
076
077        // Split message into segments
078        int[] segs = HL7Util.find(msg, SEGMENT_TERMINATOR);
079
080        // Get MSH segment with message-specific separator characters
081        MSHSegment msh = new MSHSegment(msg.substring(0, segs[0]));
082        this.segments = new HL7SegmentList(msh);
083        HL7Seps seps = msh.getHL7Seps();
084
085        // Workaround McKesson HBOC bug (extra mid-segment carriage returns)
086        for (int i = 1; i < segs.length - 1; ) {
087            if (segs[i] + 3 >= msg.length()
088              || msg.charAt(segs[i] + 1) == seps.getFieldSep()
089              || msg.charAt(segs[i] + 2) == seps.getFieldSep()
090              || msg.charAt(segs[i] + 3) == seps.getFieldSep()) {
091                msg = msg.substring(0, segs[i]) + msg.substring(segs[i] + 1);
092                for (int j = i + 1; j < segs.length; j++)
093                    segs[j]--;
094                int[] segs2 = new int[segs.length - 1];
095                System.arraycopy(segs, 0, segs2, 0, i);
096                System.arraycopy(segs, i + 1, segs2, i, segs2.length - i);
097                segs = segs2;
098            } else
099                i++;
100        }
101
102        // Add subsequent segments
103        for (int i = 0; i < segs.length - 1; i++)
104            this.segments.add(new HL7Segment(msg.substring(segs[i] + 1, segs[i + 1]), seps));
105    }
106
107    /**
108     * Get the MSH segment of this message.
109     *
110     * @return this message's MSH segment (not a copy; changes are reflected back)
111     */
112    public MSHSegment getMSHSegment() {
113        return (MSHSegment)this.segments.get(0);
114    }
115
116    /**
117     * Get all of the segments in this message.
118     *
119     * <p>
120     * The returned list does not allow removing the first element (the MSH segment) or changing it to anything other than an
121     * {@link MSHSegment}.
122     *
123     * @return list of all message segments including MSH
124     */
125    public HL7SegmentList getSegments() {
126        return this.segments;
127    }
128
129    /**
130     * Get the first occurrence of a <code>segname</code> segment in this message.
131     *
132     * @param segname segment name, e.g., "PV1"
133     * @param segnum starting segment index to start search;
134     *  index zero refers to the MSH segment
135     * @return first matching segment found, or null if not found
136     * @throws IllegalArgumentException if segnum is negative
137     */
138    public HL7Segment findSegment(String segname, int segnum) {
139        if (segnum < 0)
140            throw new IllegalArgumentException("segnum=" + segnum);
141        for (int i = segnum; i < this.segments.size(); i++) {
142            HL7Segment seg = this.segments.get(i);
143            if (seg.getName().equals(segname))
144                return seg;
145        }
146        return null;
147    }
148
149    /**
150     * Convenience method. Equivalent to:
151     *  <blockquote>
152     *  <code>findSegment(name, 0)</code>
153     *  <blockquote>
154     */
155    public HL7Segment findSegment(String segname) {
156        return this.findSegment(segname, 0);
157    }
158
159    /**
160     * Find a field by HL7 name. The name is of the form XYZ.N where
161     * "XYZ" is the segment name and "N" is the field number.
162     *
163     * @param name field name, e.g., "PV1.3"
164     * @param segnum starting segment index (zero to start from MSH segment)
165     * @return named HL7 field, or <code>null</code> if the named field doesn't exist in this message
166     * @throws IllegalArgumentException if name is not properly formatted
167     * @throws IllegalArgumentException if segnum is negative
168     */
169    public HL7Field getField(String name, int segnum) {
170
171        // Check parameters
172        Matcher matcher = FIELD_NAME_PATTERN.matcher(name);
173        if (!matcher.matches())
174            throw new IllegalArgumentException("invalid name `" + name + "'");
175        int fieldIndex;
176        try {
177            fieldIndex = Integer.parseInt(name.substring(4), 10);
178        } catch (NumberFormatException e) {
179            throw new IllegalArgumentException("invalid name `" + name + "'");
180        }
181
182        // Find segment, then field
183        HL7Segment seg = this.findSegment(name.substring(0, 3), segnum);
184        return seg != null ? seg.getField(fieldIndex) : null;
185    }
186
187    /**
188     * Convenience method. Equivalent to:
189     *  <blockquote>
190     *  <code>getField(name, 0)</code>
191     *  <blockquote>
192     */
193    public HL7Field getField(String name) {
194        return this.getField(name, 0);
195    }
196
197    /**
198     * Find a string value by HL7 name. The name is of the form {@code XYZ.N[.M[.L]]}
199     * where {@code XYZ} is the segment name, {@code N} is the field number, {@code M} is the
200     * optional component number, and {@code L} is the optional subcomponent number.
201     *
202     * <p>
203     * If either of {@code M} or {@code L} are not given, they are assumed to be {@code 1}
204     * so an unambiguous string value can always be returned.
205     *
206     * @param name HL7 value name, e.g., "PV1.3", "MSH.9.1", "ZZZ.3.2.1"
207     * @param segnum starting segment index
208     * @param repeat repeat index (starting from zero)
209     * @return named HL7 value, or <code>null</code> if the named value doesn't exist in this message
210     * @throws IllegalArgumentException if name is not properly formatted
211     * @throws IllegalArgumentException if "M" or "L" is negative
212     * @throws IllegalArgumentException if segnum or repeat is negative
213     */
214    public String get(String name, int segnum, int repeat) {
215
216        // Find field first
217        Matcher matcher = VALUE_NAME_PATTERN.matcher(name);
218        if (!matcher.matches())
219            throw new IllegalArgumentException("invalid name `" + name + "'");
220        HL7Field field = this.getField(matcher.group(1), segnum);
221        if (field == null)
222            return null;
223
224        // Get component and subcomponent indicies
225        int component = 0;
226        String value = matcher.group(4);
227        if (value != null) {
228            try {
229                component = Integer.parseInt(value, 10) - 1;
230            } catch (NumberFormatException e) {
231                throw new IllegalArgumentException("invalid component index `" + value + "'");
232            }
233        }
234        int subcomponent = 0;
235        value = matcher.group(6);
236        if (value != null) {
237            try {
238                subcomponent = Integer.parseInt(value, 10) - 1;
239            } catch (NumberFormatException e) {
240                throw new IllegalArgumentException("invalid subcomponent index `" + value + "'");
241            }
242        }
243
244        // Return value (if any)
245        return field.get(repeat, component, subcomponent);
246    }
247
248    /**
249     * Convenience method. Equivalent to:
250     *  <blockquote>
251     *  <code>getField(name, 0, 0)</code>
252     *  <blockquote>
253     */
254    public String get(String name) {
255        return this.get(name, 0, 0);
256    }
257
258    @Override
259    public boolean equals(Object obj) {
260        if (obj == null && obj.getClass() != getClass())
261            return false;
262        HL7Message that = (HL7Message)obj;
263        return this.segments.equals(that.segments);
264    }
265
266    @Override
267    public int hashCode() {
268        return this.segments.hashCode();
269    }
270
271    /**
272     * Convert this message into a string by concatenating the segments in string form, each terminated with a carriage return.
273     */
274    @Override
275    public String toString() {
276        return this.toString(getMSHSegment().getHL7Seps());
277    }
278
279    /**
280     * Format this message using the supplied separators instead of the ones defined by the MSH segment. If <code>seps</code>
281     * does not define an escape character, then characters that need to be escaped are silently elided.
282     */
283    public String toString(HL7Seps seps) {
284        StringBuilder buf = new StringBuilder();
285        for (HL7Segment segment : this.segments) {
286            segment.append(buf, seps);
287            buf.append(SEGMENT_TERMINATOR);
288        }
289        return buf.toString();
290    }
291}
