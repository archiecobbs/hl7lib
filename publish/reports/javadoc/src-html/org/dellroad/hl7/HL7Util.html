006package org.dellroad.hl7;
007
008/**
009 * Utility methods.
010 */
011public final class HL7Util {
012
013    private HL7Util() {
014    }
015
016    /**
017     * Convenience method. Equivalent to:
018     *  <blockquote>
019     *  <code>find(value, sep, 0, sep.length())</code>
020     *  </blockquote>
021     */
022    public static int[] find(String value, char sep) {
023        return find(value, sep, 0, value.length());
024    }
025
026    /**
027     * Convenience method. Equivalent to:
028     *  <blockquote>
029     *  <code>find(value, sep, start, sep.length())</code>
030     *  </blockquote>
031     */
032    public static int[] find(String value, char sep, int start) {
033        return find(value, sep, start, value.length());
034    }
035
036    /**
037     * Find all occurrences of the separator character in the sub-string.
038     *
039     * @param value entire string
040     * @param sep separator character to search for
041     * @param start starting index of sub-string to search (inclusive)
042     * @param end ending index of sub-string to search (exclusive)
043     * @return indexes of all occurrences of <code>sep</code>, in order, plus one extra index equal to <code>end</code>
044     * @throws StringIndexOutOfBoundsException if <code>start</code> is less than zero or <code>end</code> is
045     *  greater than the length of <code>value</code>
046     */
047    public static int[] find(String value, char sep, int start, int end) {
048
049        // Count occurrences
050        int count = 0;
051        for (int i = start; i < end; i++) {
052            if (value.charAt(i) == sep)
053                count++;
054        }
055
056        // Allocate array and record them
057        int[] positions = new int[count + 1];
058        if (count > 0) {
059            int j = 0;
060            for (int i = start; i < end; i++) {
061                if (value.charAt(i) == sep) {
062                    positions[j++] = i;
063                    if (j == count)
064                        break;
065                }
066            }
067        }
068
069        // Done
070        positions[count] = end;
071        return positions;
072    }
073}
