006package org.dellroad.hl7;
007
008import java.io.Serializable;
009import java.util.Arrays;
010
011/**
012 * Represents one field in an HL7 message, possibly with repeats.
013 *
014 * <p>
015 * Instances of this class are "immutable"; however, for this to be
016 * true the calling application must not alter the arrays passed to
017 * the constructor or returned from any of the methods.
018 */
019@SuppressWarnings("serial")
020public final class HL7Field implements Serializable {
021
022    /**
023     * The empty field.
024     */
025    public static final HL7Field EMPTY = new HL7Field("");
026
027    protected final String[][][] value;
028
029    /**
030     * Constructor.
031     *
032     * @param value simple string value for this field; null is treated like the empty string
033     */
034    public HL7Field(String value) {
035        this(new String[][][] { { { value } } });
036    }
037
038    /**
039     * Constructor used to create a field containing zero or more repeated values, each a simple string value.
040     *
041     * @param repeats repeat values for this field; null is treated like the empty string
042     * @throws IllegalArgumentException if <code>repeats</code> array has length zero
043     */
044    public HL7Field(String[] repeats) {
045        this.value = new String[repeats.length][1][1];
046        for (int i = 0; i < repeats.length; i++)
047            this.value[i][0][0] = repeats[i];
048        this.checkLengthsAndReplaceNulls();
049    }
050
051    /**
052     * Constructor used to create a field containing zero or more repeated values, each containing zero or more components,
053     * each component containing zero or more sub-components.
054     *
055     * @param repeats repeat array, each containing a component array, each containing a sub-component array; null in any
056     *  String position is treated like the empty string
057     * @throws NullPointerException if any intermediate array element (i.e., not of type String) is null
058     * @throws IllegalArgumentException if <code>repeats</code> array or any sub-array has length zero
059     */
060    public HL7Field(String[][][] repeats) {
061        this.value = repeats;
062        this.checkLengthsAndReplaceNulls();
063    }
064
065    /**
066     * Parsing constructor.
067     *
068     * <p>
069     * Parses the encoded HL7 field. Non-custom escapes will be decoded, while custom escapes will be silently removed.
070     *
071     * @param field encoded HL7 field contents
072     * @param seps HL7 separator characters
073     */
074    public HL7Field(String field, HL7Seps seps) {
075        int[] repSeps = HL7Util.find(field, seps.getRepSep());
076        this.value = new String[repSeps.length][][];
077        int repPosn = 0;
078        for (int i = 0; i < repSeps.length; i++) {
079            int[] compSeps = HL7Util.find(field, seps.getCompSep(), repPosn, repSeps[i]);
080            this.value[i] = new String[compSeps.length][];
081            int compPosn = repPosn;
082            for (int j = 0; j < compSeps.length; j++) {
083                if (!seps.hasSubcomponentSeparator()) {
084                    this.value[i][j] = new String[] { seps.unescape(field.substring(compPosn, compSeps[j])) };
085                    continue;
086                }
087                int[] subSeps = HL7Util.find(field, seps.getSubSep(), compPosn, compSeps[j]);
088                this.value[i][j] = new String[subSeps.length];
089                int subPosn = compPosn;
090                for (int k = 0; k < subSeps.length; k++) {
091                    this.value[i][j][k] = seps.unescape(field.substring(subPosn, subSeps[k]));
092                    subPosn = subSeps[k] + 1;
093                }
094                compPosn = compSeps[j] + 1;
095            }
096            repPosn = repSeps[i] + 1;
097        }
098    }
099
100    /**
101     * Copy constructor.
102     *
103     * <p>
104     * This constructor performs a deep copy of <code>field</code>, so that subsequent changes to it do not affect this instance.
105     */
106    public HL7Field(HL7Field field) {
107        this.value = field.value.clone();
108        for (int i = 0; i < this.value.length; i++) {
109            this.value[i] = this.value[i].clone();
110            for (int j = 0; j < this.value.length; j++)
111                this.value[i][j] = this.value[i][j].clone();
112        }
113    }
114
115    /**
116     * Get field contents.
117     *
118     * <p>
119     * The length of the returned array, each of its array elements, and each of their sub-array elements,
120     * is guaranteed to be at least one.
121     *
122     * @return array of repeats, each an array of components, each an array of sub-components, each a non-null String.
123     */
124    public String[][][] getValue() {
125        return this.value;
126    }
127
128    public boolean isEmpty() {
129        return EMPTY.equals(this);
130    }
131
132    /**
133     * Get a specific field sub-component.
134     *
135     * @param repnum repeat number (zero-based)
136     * @param compnum component number (zero-based)
137     * @param subnum sub-component number (zero-based)
138     * @return specified sub-component, or null if it does not exist
139     * @throws IllegalArgumentException if any parameter is negative
140     */
141    public String get(int repnum, int compnum, int subnum) {
142        if (repnum < 0 || compnum < 0 || subnum < 0)
143            throw new IllegalArgumentException("negative index");
144        try {
145            return this.value[repnum][compnum][subnum];
146        } catch (ArrayIndexOutOfBoundsException e) {
147            return null;
148        }
149    }
150
151    /**
152     * Append properly separated and escaped version of this field
153     * to the given buffer.
154     *
155     * @param seps HL7 separator and escape characters
156     */
157    public void append(StringBuilder buf, HL7Seps seps) {
158        for (int i = 0; i < this.value.length; i++) {
159            if (i > 0)
160                buf.append(seps.getRepSep());
161            for (int j = 0; j < this.value[i].length; j++) {
162                if (j > 0)
163                    buf.append(seps.getCompSep());
164                for (int k = 0; k < this.value[i][j].length; k++) {
165                    if (k > 0)
166                        buf.append(seps.getSubSep());
167                    seps.escape(this.value[i][j][k], buf);
168                }
169            }
170        }
171    }
172
173    /**
174     * Convert to a string using the provided separators.
175     *
176     * @param seps HL7 separator and escape characters
177     * @return this field properly separated and escaped with <code>seps</code>
178     */
179    public String toString(HL7Seps seps) {
180        StringBuilder buf = new StringBuilder();
181        this.append(buf, seps);
182        return buf.toString();
183    }
184
185    /**
186     * Convert to a string using {@link HL7Seps#DEFAULT}.
187     */
188    @Override
189    public String toString() {
190        return this.toString(HL7Seps.DEFAULT);
191    }
192
193    @Override
194    public boolean equals(Object obj) {
195        if (!(obj instanceof HL7Field))
196            return false;
197        HL7Field that = (HL7Field)obj;
198        return Arrays.deepEquals(this.value, that.value);
199    }
200
201    @Override
202    public int hashCode() {
203        return Arrays.deepHashCode(this.value);
204    }
205
206    private void checkLengthsAndReplaceNulls() {
207        if (this.value.length == 0)
208            throw new IllegalArgumentException("zero length array");
209        for (int i = 0; i < this.value.length; i++) {
210            if (this.value[i].length == 0)
211                throw new IllegalArgumentException("zero length sub-array");
212            for (int j = 0; j < this.value[i].length; j++) {
213                if (this.value[i][j].length == 0)
214                    throw new IllegalArgumentException("zero length sub-array");
215                for (int k = 0; k < this.value[i][j].length; k++) {
216                    String s = this.value[i][j][k];
217                    this.value[i][j][k] = s != null ? s : "";
218                }
219            }
220        }
221    }
222}
