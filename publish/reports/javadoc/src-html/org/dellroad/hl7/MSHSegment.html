011/**
012 * Represents an MSH segment.
013 *
014 * <p>
015 * This subclass does not allow accessing MSH.1 or MSH.2 except
016 * indirectly via {@link #getHL7Seps} and {@link #setHL7Seps setHL7Seps()}.
017 */
018@SuppressWarnings("serial")
019public class MSHSegment extends HL7Segment {
020
021    public static final String DEFAULT_PROCESSING_ID = "P";
022    public static final String DEFAULT_VERSION_ID = "2.3";
023
024    public static final String MSH_SEGMENT_NAME = "MSH";
025    public static final String MSA_SEGMENT_NAME = "MSA";
026
027    public static final String TIMESTAMP_FORMAT = "yyyyMMddHHmmss.SSSZ";
028
029    private static final String ACK = "ACK";
030    private static final HL7Field ACK_MSH_9 = new HL7Field(ACK);
031    private static final HL7Field ACK_MSA_1 = new HL7Field("AA");
032
033    /**
034     * Parsing constructor.
035     *
036     * @param line encoded HL7 MSH segment
037     * @throws HL7ContentException if segment is invalid
038     */
039    public MSHSegment(String line) throws HL7ContentException {
040
041        // Parse "MSH", MSH.1, and MSH.2
042        if (!line.startsWith(MSH_SEGMENT_NAME))
043            throw new HL7ContentException("MSH segment does not start with `" + MSH_SEGMENT_NAME + "'").setContent(line);
044        this.fields.add(new HL7Field(MSH_SEGMENT_NAME));
045        if (line.length() < 6)
046            throw new HL7ContentException("MSH segment is truncated").setContent(line);
047        char fieldSep = line.charAt(3);
048        char repSep = line.charAt(5);
049        char compSep = line.charAt(4);
050        char subSep = '\u0000';
051        char escChar = '\u0000';
052        int fieldSep23 = 6;
053        if (line.length() >= 7 && line.charAt(6) != fieldSep) {
054            fieldSep23++;
055            escChar = line.charAt(6);
056            if (line.length() >= 8 && line.charAt(7) != fieldSep) {
057                fieldSep23++;
058                subSep = line.charAt(7);
059            }
060        }
061        HL7Seps seps = new HL7Seps(fieldSep, compSep, repSep, escChar, subSep);
062        this.setHL7Seps(seps);
063
064        // Parse remaining fields (if any)
065        if (fieldSep23 == line.length())
066            return;
067        if (line.charAt(fieldSep23) != fieldSep)
068            throw new HL7ContentException("bogus extra characters in MSH.2").setContent(line);
069        this.parseAndAddFields(line.substring(fieldSep23 + 1), seps);
070    }

072    /**
073     * Constructor.
074     *
075     * This constructor initializes only MSH.1 and MSH.2.
076     *
077     * @param seps separator and escape characters for this message

079    public MSHSegment(HL7Seps seps) {
080        this.fields.add(new HL7Field(MSH_SEGMENT_NAME));
081        this.setHL7Seps(seps);
082    }

209    /**
210     * Get the separator and escape characters defined by MSH.1 and MSH.2.
211     */
212    public HL7Seps getHL7Seps() {
213        String msh1 = this.getField(1).get(0, 0, 0);
214        String msh2 = this.getField(2).get(0, 0, 0);
215        try {
216            return new HL7Seps(msh1.charAt(0), msh2.charAt(0), msh2.charAt(1),
217              msh2.length() > 2 ? msh2.charAt(2) : '\u0000',
218              msh2.length() > 3 ? msh2.charAt(3) : '\u0000');
219        } catch (HL7ContentException e) {
220            throw new RuntimeException("impossible", e);
221        }
222    }
223
224    /**
225     * Set MSH.1 and MSH.2 based on the provided separator and
226     * escape characters.
227     */
228    public void setHL7Seps(HL7Seps seps) {
229        super.setField(1, new HL7Field("" + seps.getFieldSep()));
230        super.setField(2, new HL7Field("" + seps.getCompSep() + seps.getRepSep()
231          + (seps.hasEscapeCharacter() ? "" + seps.getEscChar() : "")
232          + (seps.hasSubcomponentSeparator() ? "" + seps.getSubSep() : "")));
233    }

260    /**
261     * Create an acknowlegement message for this message.
262     *
263     * @param serno next local HL7 serial number
264     * @return an acknowlegement of this message
265     * @throws HL7ContentException if the required fields are not present
266     *  in this instance
267     */
268    public HL7Message createACK(long serno) throws HL7ContentException {
269
270        // We must have at least up to MSH.11
271        if (getNumFields() < 12)
272            throw new HL7ContentException("insufficient fields for ACK'ing");
273        HL7Field versionID = this.getVersionID() != null ? this.getVersionID() : new HL7Field(DEFAULT_VERSION_ID);
274
275        // Build ACK
276        HL7Message ack = new HL7Message();
277        MSHSegment msh = ack.getMSHSegment();
278        msh.setTimestamp(new HL7Field(new SimpleDateFormat(TIMESTAMP_FORMAT).format(new Date())));
279        msh.setMessageType(ACK_MSH_9);
280        msh.setControlID(new HL7Field("" + serno));
281        msh.setProcessingID(this.getProcessingID());
282        msh.setVersionID(versionID);
283        HL7Segment msa = new HL7Segment(MSA_SEGMENT_NAME);
284        msa.setField(1, ACK_MSA_1);
285        msa.setField(2, this.getControlID());
286        ack.getSegments().add(msa);
287
288        // Done
289        return ack;
290    }

292    /**
293     * Check whether the given message is an acknowlegement of this MSH.
294     *
295     * @param ack putative acknowlegement of this message
296     * @throws HL7ContentException if the required fields are not present
297     *  in this instance
298     */
299    public boolean isACK(HL7Message ack) throws HL7ContentException {
300        HL7Field controlID = this.getControlID();
301        HL7Field procID = this.getProcessingID();
302        if (controlID == null || procID == null)
303            throw new HL7ContentException("insufficient fields for ACK'ing");
304        MSHSegment ackMSH = ack.getMSHSegment();
305        return ACK.equals(ack.getField("MSH.9").get(0, 0, 0))
306          && controlID.equals(ack.getField("MSA.2"))
307          && procID.equals(ackMSH.getProcessingID())
308          && ACK_MSA_1.equals(ack.getField("MSA.1"));
309    }

311    /**
312     * Append string encoding of this segment to the provided buffer.
313     *
314     * <p>
315     * This overridden version ensures that MSH.1 and MSH.2 are not escaped.
316     */
317    @Override
318    public void append(StringBuilder buf, HL7Seps seps) {
319        HL7Field[] fields = this.fields.toArray(new HL7Field[this.fields.size()]);
320        buf.append(fields[0]);
321        buf.append(seps);
322        for (int i = 3; i < fields.length; i++) {
323            buf.append(seps.getFieldSep());
324            fields[i].append(buf, seps);
325        }
326    }

328    /**
329     * Convert to a string using the provided separators.
330     *
331     * <p>
332     * This overridden version ensures that MSH.1 and MSH.2 are consistent with <code>seps</code> in the generated string.
333     * This separators configured in this instance are not affected regardless.
334     */
335    @Override
336    public String toString(HL7Seps seps) {
337        if (seps.equals(getHL7Seps()))
338            return super.toString(seps);
339        MSHSegment copy = new MSHSegment(this);
340        copy.setHL7Seps(seps);
341        return copy.toString();
342    }
343
344    @Override
345    public String toString() {
346        return super.toString(getHL7Seps());
347    }
