006package org.dellroad.hl7;
007
008import java.io.IOException;
009
010/**
011 * Input source for HL7 messages.
012 */
013public interface HL7Reader {
014
015    /**
016     * Read the next HL7 message.
017     *
018     * @throws HL7ContentException if an invalid HL7 message is received, decoded, etc. by this instance
019     * @throws java.io.EOFException if there is no more input
020     * @throws IOException if some other error occurs
021     * @throws IOException if {@link #close} has already been invoked
022     */
023    HL7Message readMessage() throws IOException, HL7ContentException;
024
025    /**
026     * Close this instance.
027     */
028    void close() throws IOException;
029}
