008    package org.dellroad.hl7;
009    
010    /**
011     * Utility methods.
012     */
013    public final class HL7Util {
014    
015        private HL7Util() {
016        }
017    
018        /**
019         * Convenience method. Equivalent to:
020         *  <blockquote>
021         *  <code>find(value, sep, 0, sep.length())</code>
022         *  </blockquote>
023         */
024        public static int[] find(String value, char sep) {
025            return find(value, sep, 0, value.length());
026        }
027    
028        /**
029         * Convenience method. Equivalent to:
030         *  <blockquote>
031         *  <code>find(value, sep, start, sep.length())</code>
032         *  </blockquote>
033         */
034        public static int[] find(String value, char sep, int start) {
035            return find(value, sep, start, value.length());
036        }
037    
038        /**
039         * Find all occurrences of the separator character in the sub-string.
040         *
041         * @param value entire string
042         * @param sep separator character to search for
043         * @param start starting index of sub-string to search (inclusive)
044         * @param end ending index of sub-string to search (exclusive)
045         * @return indexes of all occurrences of <code>sep</code>, in order, plus one extra index equal to <code>end</code>
046         * @throws StringIndexOutOfBoundsException if <code>start</code> is less than zero or <code>end</code> is
047         *  greater than the length of <code>value</code>
048         */
049        public static int[] find(String value, char sep, int start, int end) {
050    
051            // Count occurrences
052            int count = 0;
053            for (int i = start; i < end; i++) {
054                if (value.charAt(i) == sep)
055                    count++;
056            }
057    
058            // Allocate array and record them
059            int[] positions = new int[count + 1];
060            if (count > 0) {
061                int j = 0;
062                for (int i = start; i < end; i++) {
063                    if (value.charAt(i) == sep) {
064                        positions[j++] = i;
065                        if (j == count)
066                            break;
067                    }
068                }
069            }
070    
071            // Done
072            positions[count] = end;
073            return positions;
074        }
075    }
