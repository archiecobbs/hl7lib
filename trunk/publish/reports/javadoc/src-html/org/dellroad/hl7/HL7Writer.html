008    package org.dellroad.hl7;
009    
010    import java.io.IOException;
011    
012    /**
013     * Implemented by classes whose instances are capable of sending out HL7 messages.
014     */
015    public interface HL7Writer {
016    
017        /**
018         * Write a message.
019         *
020         * @throws IOException if there is some error
021         * @throws IOException if {@link #close} has already been invoked
022         */
023        void writeMessage(HL7Message message) throws IOException;
024    
025        /**
026         * Close this instance.
027         */
028        void close() throws IOException;
029    }
