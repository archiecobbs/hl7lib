008    package org.dellroad.hl7;
009    
010    import java.io.Serializable;
011    import java.util.Arrays;
012    
013    /**
014     * Represents one field in an HL7 message, possibly with repeats.
015     *
016     * <p>
017     * Instances of this class are "immutable"; however, for this to be
018     * true the calling application must not alter the arrays passed to
019     * the constructor or returned from any of the methods.
020     */
021    @SuppressWarnings("serial")
022    public final class HL7Field implements Serializable {
023    
024        /**
025         * The empty field.
026         */
027        public static final HL7Field EMPTY = new HL7Field("");
028    
029        protected final String[][][] value;
030    
031        /**
032         * Constructor.
033         *
034         * @param value simple string value for this field; null is treated like the empty string
035         */
036        public HL7Field(String value) {
037            this(new String[][][] { { { value } } });
038        }
039    
040        /**
041         * Constructor used to create a field containing zero or more repeated values, each a simple string value.
042         *
043         * @param repeats repeat values for this field; null is treated like the empty string
044         * @throws IllegalArgumentException if <code>repeats</code> array has length zero
045         */
046        public HL7Field(String[] repeats) {
047            this.value = new String[repeats.length][1][1];
048            for (int i = 0; i < repeats.length; i++)
049                this.value[i][0][0] = repeats[i];
050            this.checkLengthsAndReplaceNulls();
051        }
052    
053        /**
054         * Constructor used to create a field containing zero or more repeated values, each containing zero or more components,
055         * each component containing zero or more sub-components.
056         *
057         * @param repeats repeat array, each containing a component array, each containing a sub-component array; null in any
058         *  String position is treated like the empty string
059         * @throws NullPointerException if any intermediate array element (i.e., not of type String) is null
060         * @throws IllegalArgumentException if <code>repeats</code> array or any sub-array has length zero
061         */
062        public HL7Field(String[][][] repeats) {
063            this.value = repeats;
064            this.checkLengthsAndReplaceNulls();
065        }
066    
067        /**
068         * Parsing constructor.
069         *
070         * <p>
071         * Parses the encoded HL7 field. Non-custom escapes will be decoded, while custom escapes will be silently removed.
072         *
073         * @param field encoded HL7 field contents
074         * @param seps HL7 separator characters
075         */
076        public HL7Field(String field, HL7Seps seps) {
077            int[] repSeps = HL7Util.find(field, seps.getRepSep());
078            this.value = new String[repSeps.length][][];
079            int repPosn = 0;
080            for (int i = 0; i < repSeps.length; i++) {
081                int[] compSeps = HL7Util.find(field, seps.getCompSep(), repPosn, repSeps[i]);
082                this.value[i] = new String[compSeps.length][];
083                int compPosn = repPosn;
084                for (int j = 0; j < compSeps.length; j++) {
085                    if (!seps.hasSubcomponentSeparator()) {
086                        this.value[i][j] = new String[] { seps.unescape(field.substring(compPosn, compSeps[j])) };
087                        continue;
088                    }
089                    int[] subSeps = HL7Util.find(field, seps.getSubSep(), compPosn, compSeps[j]);
090                    this.value[i][j] = new String[subSeps.length];
091                    int subPosn = compPosn;
092                    for (int k = 0; k < subSeps.length; k++) {
093                        this.value[i][j][k] = seps.unescape(field.substring(subPosn, subSeps[k]));
094                        subPosn = subSeps[k] + 1;
095                    }
096                    compPosn = compSeps[j] + 1;
097                }
098                repPosn = repSeps[i] + 1;
099            }
100        }
101    
102        /**
103         * Copy constructor.
104         *
105         * <p>
106         * This constructor performs a deep copy of <code>field</code>, so that subsequent changes to it do not affect this instance.
107         */
108        public HL7Field(HL7Field field) {
109            this.value = field.value.clone();
110            for (int i = 0; i < this.value.length; i++) {
111                this.value[i] = this.value[i].clone();
112                for (int j = 0; j < this.value.length; j++)
113                    this.value[i][j] = this.value[i][j].clone();
114            }
115        }
116    
117        /**
118         * Get field contents.
119         *
120         * <p>
121         * The length of the returned array, each of its array elements, and each of their sub-array elements,
122         * is guaranteed to be at least one.
123         *
124         * @return array of repeats, each an array of components, each an array of sub-components, each a non-null String.
125         */
126        public String[][][] getValue() {
127            return this.value;
128        }
129    
130        public boolean isEmpty() {
131            return EMPTY.equals(this);
132        }
133    
134        /**
135         * Get a specific field sub-component.
136         *
137         * @param repnum repeat number (zero-based)
138         * @param compnum component number (zero-based)
139         * @param subnum sub-component number (zero-based)
140         * @return specified sub-component, or null if it does not exist
141         * @throws IllegalArgumentException if any parameter is negative
142         */
143        public String get(int repnum, int compnum, int subnum) {
144            if (repnum < 0 || compnum < 0 || subnum < 0)
145                throw new IllegalArgumentException("negative index");
146            try {
147                return this.value[repnum][compnum][subnum];
148            } catch (ArrayIndexOutOfBoundsException e) {
149                return null;
150            }
151        }
152    
153        /**
154         * Append properly separated and escaped version of this field
155         * to the given buffer.
156         *
157         * @param seps HL7 separator and escape characters
158         */
159        public void append(StringBuilder buf, HL7Seps seps) {
160            for (int i = 0; i < this.value.length; i++) {
161                if (i > 0)
162                    buf.append(seps.getRepSep());
163                for (int j = 0; j < this.value[i].length; j++) {
164                    if (j > 0)
165                        buf.append(seps.getCompSep());
166                    for (int k = 0; k < this.value[i][j].length; k++) {
167                        if (k > 0)
168                            buf.append(seps.getSubSep());
169                        seps.escape(this.value[i][j][k], buf);
170                    }
171                }
172            }
173        }
174    
175        /**
176         * Convert to a string using the provided separators.
177         *
178         * @param seps HL7 separator and escape characters
179         * @return this field properly separated and escaped with <code>seps</code>
180         */
181        public String toString(HL7Seps seps) {
182            StringBuilder buf = new StringBuilder();
183            this.append(buf, seps);
184            return buf.toString();
185        }
186    
187        /**
188         * Convert to a string using {@link HL7Seps#DEFAULT}.
189         */
190        @Override
191        public String toString() {
192            return this.toString(HL7Seps.DEFAULT);
193        }
194    
195        @Override
196        public boolean equals(Object obj) {
197            if (!(obj instanceof HL7Field))
198                return false;
199            HL7Field that = (HL7Field)obj;
200            return Arrays.deepEquals(this.value, that.value);
201        }
202    
203        @Override
204        public int hashCode() {
205            return Arrays.deepHashCode(this.value);
206        }
207    
208        private void checkLengthsAndReplaceNulls() {
209            if (this.value.length == 0)
210                throw new IllegalArgumentException("zero length array");
211            for (int i = 0; i < this.value.length; i++) {
212                if (this.value[i].length == 0)
213                    throw new IllegalArgumentException("zero length sub-array");
214                for (int j = 0; j < this.value[i].length; j++) {
215                    if (this.value[i][j].length == 0)
216                        throw new IllegalArgumentException("zero length sub-array");
217                    for (int k = 0; k < this.value[i][j].length; k++) {
218                        String s = this.value[i][j][k];
219                        this.value[i][j][k] = s != null ? s : "";
220                    }
221                }
222            }
223        }
224    }
