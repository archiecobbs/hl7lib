008package org.dellroad.hl7;
009
010import java.io.Serializable;
011import java.util.regex.Matcher;
012import java.util.regex.Pattern;
013
014/**
015 * Represents an HL7 message.
016 */
017@SuppressWarnings("serial")
018public class HL7Message implements Serializable {
019
020    /**
021     * The character that separates segments in an HL7 message.
022     */
023    public static final char SEGMENT_TERMINATOR = '\r';
024
025    private static final Pattern FIELD_NAME_PATTERN = Pattern.compile("([\\p{Alnum}]{3})\\.[0-9]+");
026
027    private static final Pattern VALUE_NAME_PATTERN = Pattern.compile("(" + FIELD_NAME_PATTERN + ")(\\.([0-9]+)(\\.([0-9]+))?)?");
028
029    /**
030     * The segments in this message. This list always contains at least
031     * one element, namely the {@link MSHSegment}.
032     */
033    protected final HL7SegmentList segments;
034
035    /**
036     * Construct a new HL7 message containing only the given MSH segment.
037     *
038     * @param msh MSH segment
039     */
040    public HL7Message(MSHSegment msh) {
041        this.segments = new HL7SegmentList(msh);
042    }
043
044    /**
045     * Construct an empty HL7 message.
046     *
047     * <p>
048     * A MSH segment will be automatically added.
049     *
050     * @param seps separator and escape characters to use for this message
051     */
052    public HL7Message(HL7Seps seps) {
053        this.segments = new HL7SegmentList(new MSHSegment(seps));
054    }
055
056    /**
057     * Convenience constructor. Equivalent to:
058     *  <blockquote>
059     *  <code>HL7Message(HL7Seps.DEFAULT)</code>
060     *  <blockquote>
061     */
062    public HL7Message() {
063        this(HL7Seps.DEFAULT);
064    }
065
066    /**
067     * Parsing constructor. Constructs an HL7 message by parsing the given string.
068     * Segments must be separated with a carriage return character.
069     *
070     * @throws HL7ContentException if the string is invalid
071     */
072    public HL7Message(String msg) throws HL7ContentException {
073
074        // Eliminate trailing CR's
075        int len = msg.length();
076        while (len > 0 && msg.charAt(len - 1) == SEGMENT_TERMINATOR)
077            msg = msg.substring(0, --len);
078
079        // Split message into segments
080        int[] segs = HL7Util.find(msg, SEGMENT_TERMINATOR);
081
082        // Get MSH segment with message-specific separator characters
083        MSHSegment msh = new MSHSegment(msg.substring(0, segs[0]));
084        this.segments = new HL7SegmentList(msh);
085        HL7Seps seps = msh.getHL7Seps();
086
087        // Workaround McKesson HBOC bug (extra mid-segment carriage returns)
088        for (int i = 1; i < segs.length - 1; ) {
089            if (segs[i] + 3 >= msg.length()
090              || msg.charAt(segs[i] + 1) == seps.getFieldSep()
091              || msg.charAt(segs[i] + 2) == seps.getFieldSep()
092              || msg.charAt(segs[i] + 3) == seps.getFieldSep()) {
093                msg = msg.substring(0, segs[i]) + msg.substring(segs[i] + 1);
094                for (int j = i + 1; j < segs.length; j++)
095                    segs[j]--;
096                int[] segs2 = new int[segs.length - 1];
097                System.arraycopy(segs, 0, segs2, 0, i);
098                System.arraycopy(segs, i + 1, segs2, i, segs2.length - i);
099                segs = segs2;
100            } else
101                i++;
102        }
103
104        // Add subsequent segments
105        for (int i = 0; i < segs.length - 1; i++)
106            this.segments.add(new HL7Segment(msg.substring(segs[i] + 1, segs[i + 1]), seps));
107    }
108
109    /**
110     * Get the MSH segment of this message.
111     *
112     * @return this message's MSH segment (not a copy; changes are reflected back)
113     */
114    public MSHSegment getMSHSegment() {
115        return (MSHSegment)this.segments.get(0);
116    }
117
118    /**
119     * Get all of the segments in this message.
120     *
121     * <p>
122     * The returned list does not allow removing the first element (the MSH segment) or changing it to anything other than an
123     * {@link MSHSegment}.
124     *
125     * @return list of all message segments including MSH
126     */
127    public HL7SegmentList getSegments() {
128        return this.segments;
129    }
130
131    /**
132     * Get the first occurrence of a <code>segname</code> segment in this message.
133     *
134     * @param segname segment name, e.g., "PV1"
135     * @param segnum starting segment index to start search;
136     *  index zero refers to the MSH segment
137     * @return first matching segment found, or null if not found
138     * @throws IllegalArgumentException if segnum is negative
139     */
140    public HL7Segment findSegment(String segname, int segnum) {
141        if (segnum < 0)
142            throw new IllegalArgumentException("segnum=" + segnum);
143        for (int i = segnum; i < this.segments.size(); i++) {
144            HL7Segment seg = this.segments.get(i);
145            if (seg.getName().equals(segname))
146                return seg;
147        }
148        return null;
149    }
150
151    /**
152     * Convenience method. Equivalent to:
153     *  <blockquote>
154     *  <code>findSegment(name, 0)</code>
155     *  <blockquote>
156     */
157    public HL7Segment findSegment(String segname) {
158        return this.findSegment(segname, 0);
159    }
160
161    /**
162     * Find a field by HL7 name. The name is of the form XYZ.N where
163     * "XYZ" is the segment name and "N" is the field number.
164     *
165     * @param name field name, e.g., "PV1.3"
166     * @param segnum starting segment index (zero to start from MSH segment)
167     * @return named HL7 field, or <code>null</code> if the named field doesn't exist in this message
168     * @throws IllegalArgumentException if name is not properly formatted
169     * @throws IllegalArgumentException if segnum is negative
170     */
171    public HL7Field getField(String name, int segnum) {
172
173        // Check parameters
174        Matcher matcher = FIELD_NAME_PATTERN.matcher(name);
175        if (!matcher.matches())
176            throw new IllegalArgumentException("invalid name `" + name + "'");
177        int fieldIndex;
178        try {
179            fieldIndex = Integer.parseInt(name.substring(4), 10);
180        } catch (NumberFormatException e) {
181            throw new IllegalArgumentException("invalid name `" + name + "'");
182        }
183
184        // Find segment, then field
185        HL7Segment seg = this.findSegment(name.substring(0, 3), segnum);
186        return seg != null ? seg.getField(fieldIndex) : null;
187    }
188
189    /**
190     * Convenience method. Equivalent to:
191     *  <blockquote>
192     *  <code>getField(name, 0)</code>
193     *  <blockquote>
194     */
195    public HL7Field getField(String name) {
196        return this.getField(name, 0);
197    }
198
199    /**
200     * Find a string value by HL7 name. The name is of the form {@code XYZ.N[.M[.L]]}
201     * where {@code XYZ} is the segment name, {@code N} is the field number, {@code M} is the
202     * optional component number, and {@code L} is the optional subcomponent number.
203     *
204     * <p>
205     * If either of {@code M} or {@code L} are not given, they are assumed to be {@code 1}
206     * so an unambiguous string value can always be returned.
207     *
208     * @param name HL7 value name, e.g., "PV1.3", "MSH.9.1", "ZZZ.3.2.1"
209     * @param segnum starting segment index
210     * @param repeat repeat index (starting from zero)
211     * @return named HL7 value, or <code>null</code> if the named value doesn't exist in this message
212     * @throws IllegalArgumentException if name is not properly formatted
213     * @throws IllegalArgumentException if "M" or "L" is negative
214     * @throws IllegalArgumentException if segnum or repeat is negative
215     */
216    public String get(String name, int segnum, int repeat) {
217
218        // Find field first
219        Matcher matcher = VALUE_NAME_PATTERN.matcher(name);
220        if (!matcher.matches())
221            throw new IllegalArgumentException("invalid name `" + name + "'");
222        HL7Field field = this.getField(matcher.group(1), segnum);
223        if (field == null)
224            return null;
225
226        // Get component and subcomponent indicies
227        int component = 0;
228        String value = matcher.group(4);
229        if (value != null) {
230            try {
231                component = Integer.parseInt(value, 10) - 1;
232            } catch (NumberFormatException e) {
233                throw new IllegalArgumentException("invalid component index `" + value + "'");
234            }
235        }
236        int subcomponent = 0;
237        value = matcher.group(6);
238        if (value != null) {
239            try {
240                subcomponent = Integer.parseInt(value, 10) - 1;
241            } catch (NumberFormatException e) {
242                throw new IllegalArgumentException("invalid subcomponent index `" + value + "'");
243            }
244        }
245
246        // Return value (if any)
247        return field.get(repeat, component, subcomponent);
248    }
249
250    /**
251     * Convenience method. Equivalent to:
252     *  <blockquote>
253     *  <code>getField(name, 0, 0)</code>
254     *  <blockquote>
255     */
256    public String get(String name) {
257        return this.get(name, 0, 0);
258    }
259
260    @Override
261    public boolean equals(Object obj) {
262        if (obj == null && obj.getClass() != getClass())
263            return false;
264        HL7Message that = (HL7Message)obj;
265        return this.segments.equals(that.segments);
266    }
267
268    @Override
269    public int hashCode() {
270        return this.segments.hashCode();
271    }
272
273    /**
274     * Convert this message into a string by concatenating the segments in string form, each terminated with a carriage return.
275     */
276    @Override
277    public String toString() {
278        return this.toString(getMSHSegment().getHL7Seps());
279    }
280
281    /**
282     * Format this message using the supplied separators instead of the ones defined by the MSH segment. If <code>seps</code>
283     * does not define an escape character, then characters that need to be escaped are silently elided.
284     */
285    public String toString(HL7Seps seps) {
286        StringBuilder buf = new StringBuilder();
287        for (HL7Segment segment : this.segments) {
288            segment.append(buf, seps);
289            buf.append(SEGMENT_TERMINATOR);
290        }
291        return buf.toString();
292    }
293}
