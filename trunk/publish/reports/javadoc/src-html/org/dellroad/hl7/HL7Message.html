008    package org.dellroad.hl7;
009    
010    import java.io.Serializable;
011    import java.util.List;
012    import java.util.regex.Matcher;
013    import java.util.regex.Pattern;
014    
015    /**
016     * Represents an HL7 message.
017     */
018    @SuppressWarnings("serial")
019    public class HL7Message implements Serializable {
020    
021        /**
022         * The character that separates segments in an HL7 message.
023         */
024        public static final char SEGMENT_TERMINATOR = '\r';
025    
026        private static final Pattern FIELD_NAME_PATTERN = Pattern.compile("([\\p{Alnum}]{3})\\.[0-9]+");
027    
028        private static final Pattern VALUE_NAME_PATTERN = Pattern.compile("(" + FIELD_NAME_PATTERN + ")(\\.([0-9]+)(\\.([0-9]+))?)?");
029    
030        /**
031         * The segments in this message. This list always contains at least
032         * one element, namely the {@link MSHSegment}.
033         */
034        protected final HL7SegmentList segments;
035    
036        /**
037         * Construct a new HL7 message containing only the given MSH segment.
038         *
039         * @param msh MSH segment
040         */
041        public HL7Message(MSHSegment msh) {
042            this.segments = new HL7SegmentList(msh);
043        }
044    
045        /**
046         * Construct an empty HL7 message.
047         *
048         * <p>
049         * A MSH segment will be automatically added.
050         *
051         * @param seps separator and escape characters to use for this message
052         */
053        public HL7Message(HL7Seps seps) {
054            this.segments = new HL7SegmentList(new MSHSegment(seps));
055        }
056    
057        /**
058         * Convenience constructor. Equivalent to:
059         *  <blockquote>
060         *  <code>HL7Message(HL7Seps.DEFAULT)</code>
061         *  <blockquote>
062         */
063        public HL7Message() {
064            this(HL7Seps.DEFAULT);
065        }
066    
067        /**
068         * Parsing constructor. Constructs an HL7 message by parsing the given string.
069         * Segments must be separated with a carriage return character.
070         *
071         * @throws HL7ContentException if the string is invalid
072         */
073        public HL7Message(String msg) throws HL7ContentException {
074    
075            // Eliminate trailing CR's
076            int len = msg.length();
077            while (len > 0 && msg.charAt(len - 1) == SEGMENT_TERMINATOR)
078                msg = msg.substring(0, --len);
079    
080            // Split message into segments
081            int[] segs = HL7Util.find(msg, SEGMENT_TERMINATOR);
082    
083            // Get MSH segment with message-specific separator characters
084            MSHSegment msh = new MSHSegment(msg.substring(0, segs[0]));
085            this.segments = new HL7SegmentList(msh);
086            HL7Seps seps = msh.getHL7Seps();
087    
088            // Workaround McKesson HBOC bug (extra mid-segment carriage returns)
089            for (int i = 1; i < segs.length - 1; ) {
090                if (segs[i] + 3 >= msg.length()
091                  || msg.charAt(segs[i] + 1) == seps.getFieldSep()
092                  || msg.charAt(segs[i] + 2) == seps.getFieldSep()
093                  || msg.charAt(segs[i] + 3) == seps.getFieldSep()) {
094                    msg = msg.substring(0, segs[i]) + msg.substring(segs[i] + 1);
095                    for (int j = i + 1; j < segs.length; j++)
096                        segs[j]--;
097                    int[] segs2 = new int[segs.length - 1];
098                    System.arraycopy(segs, 0, segs2, 0, i);
099                    System.arraycopy(segs, i + 1, segs2, i, segs2.length - i);
100                    segs = segs2;
101                } else
102                    i++;
103            }
104    
105            // Add subsequent segments
106            for (int i = 0; i < segs.length - 1; i++)
107                this.segments.add(new HL7Segment(msg.substring(segs[i] + 1, segs[i + 1]), seps));
108        }
109    
110        /**
111         * Get the MSH segment of this message.
112         *
113         * @return this message's MSH segment (not a copy; changes are reflected back)
114         */
115        public MSHSegment getMSHSegment() {
116            return (MSHSegment)this.segments.get(0);
117        }
118    
119        /**
120         * Get all of the segments in this message.
121         *
122         * <p>
123         * The returned list does not allow removing the first element (the MSH segment) or changing it to anything other than an
124         * {@link MSHSegment}.
125         *
126         * @return list of all message segments including MSH
127         */
128        public List<HL7Segment> getSegments() {
129            return this.segments;
130        }
131    
132        /**
133         * Get the first occurrence of a <code>segname</code> segment in this message.
134         *
135         * @param segname segment name, e.g., "PV1"
136         * @param segnum starting segment index to start search;
137         *  index zero refers to the MSH segment
138         * @return first matching segment found, or null if not found
139         * @throws IllegalArgumentException if segnum is negative
140         */
141        public HL7Segment findSegment(String segname, int segnum) {
142            if (segnum < 0)
143                throw new IllegalArgumentException("segnum=" + segnum);
144            for (int i = segnum; i < this.segments.size(); i++) {
145                HL7Segment seg = this.segments.get(i);
146                if (seg.getName().equals(segname))
147                    return seg;
148            }
149            return null;
150        }
151    
152        /**
153         * Convenience method. Equivalent to:
154         *  <blockquote>
155         *  <code>findSegment(name, 0)</code>
156         *  <blockquote>
157         */
158        public HL7Segment findSegment(String segname) {
159            return this.findSegment(segname, 0);
160        }
161    
162        /**
163         * Find a field by HL7 name. The name is of the form XYZ.N where
164         * "XYZ" is the segment name and "N" is the field number.
165         *
166         * @param name field name, e.g., "PV1.3"
167         * @param segnum starting segment index (zero to start from MSH segment)
168         * @return named HL7 field, or <code>null</code> if the named field doesn't exist in this message
169         * @throws IllegalArgumentException if name is not properly formatted
170         * @throws IllegalArgumentException if segnum is negative
171         */
172        public HL7Field getField(String name, int segnum) {
173    
174            // Check parameters
175            Matcher matcher = FIELD_NAME_PATTERN.matcher(name);
176            if (!matcher.matches())
177                throw new IllegalArgumentException("invalid name `" + name + "'");
178            int fieldIndex;
179            try {
180                fieldIndex = Integer.parseInt(name.substring(4), 10);
181            } catch (NumberFormatException e) {
182                throw new IllegalArgumentException("invalid name `" + name + "'");
183            }
184    
185            // Find segment, then field
186            HL7Segment seg = this.findSegment(name.substring(0, 3), segnum);
187            return seg != null ? seg.getField(fieldIndex) : null;
188        }
189    
190        /**
191         * Convenience method. Equivalent to:
192         *  <blockquote>
193         *  <code>getField(name, 0)</code>
194         *  <blockquote>
195         */
196        public HL7Field getField(String name) {
197            return this.getField(name, 0);
198        }
199    
200        /**
201         * Find a string value by HL7 name. The name is of the form {@code XYZ.N[.M[.L]]}
202         * where {@code XYZ} is the segment name, {@code N} is the field number, {@code M} is the
203         * optional component number, and {@code L} is the optional subcomponent number.
204         *
205         * <p>
206         * If either of {@code M} or {@code L} are not given, they are assumed to be {@code 1}
207         * so an unambiguous string value can always be returned.
208         *
209         * @param name HL7 value name, e.g., "PV1.3", "MSH.9.1", "ZZZ.3.2.1"
210         * @param segnum starting segment index
211         * @param repeat repeat index (starting from zero)
212         * @return named HL7 value, or <code>null</code> if the named value doesn't exist in this message
213         * @throws IllegalArgumentException if name is not properly formatted
214         * @throws IllegalArgumentException if "M" or "L" is negative
215         * @throws IllegalArgumentException if segnum or repeat is negative
216         */
217        public String get(String name, int segnum, int repeat) {
218    
219            // Find field first
220            Matcher matcher = VALUE_NAME_PATTERN.matcher(name);
221            if (!matcher.matches())
222                throw new IllegalArgumentException("invalid name `" + name + "'");
223            HL7Field field = this.getField(matcher.group(1), segnum);
224            if (field == null)
225                return null;
226    
227            // Get component and subcomponent indicies
228            int component = 0;
229            String value = matcher.group(4);
230            if (value != null) {
231                try {
232                    component = Integer.parseInt(value, 10) - 1;
233                } catch (NumberFormatException e) {
234                    throw new IllegalArgumentException("invalid component index `" + value + "'");
235                }
236            }
237            int subcomponent = 0;
238            value = matcher.group(6);
239            if (value != null) {
240                try {
241                    subcomponent = Integer.parseInt(value, 10) - 1;
242                } catch (NumberFormatException e) {
243                    throw new IllegalArgumentException("invalid subcomponent index `" + value + "'");
244                }
245            }
246    
247            // Return value (if any)
248            return field.get(repeat, component, subcomponent);
249        }
250    
251        /**
252         * Convenience method. Equivalent to:
253         *  <blockquote>
254         *  <code>getField(name, 0, 0)</code>
255         *  <blockquote>
256         */
257        public String get(String name) {
258            return this.get(name, 0, 0);
259        }
260    
261        @Override
262        public boolean equals(Object obj) {
263            if (obj == null && obj.getClass() != getClass())
264                return false;
265            HL7Message that = (HL7Message)obj;
266            return this.segments.equals(that.segments);
267        }
268    
269        @Override
270        public int hashCode() {
271            return this.segments.hashCode();
272        }
273    
274        /**
275         * Convert this message into a string by concatenating the segments in string form, each terminated with a carriage return.
276         */
277        @Override
278        public String toString() {
279            return this.toString(getMSHSegment().getHL7Seps());
280        }
281    
282        /**
283         * Format this message using the supplied separators instead of the ones defined by the MSH segment. If <code>seps</code>
284         * does not define an escape character, then characters that need to be escaped are silently elided.
285         */
286        public String toString(HL7Seps seps) {
287            StringBuilder buf = new StringBuilder();
288            for (HL7Segment segment : this.segments) {
289                segment.append(buf, seps);
290                buf.append(SEGMENT_TERMINATOR);
291            }
292            return buf.toString();
293        }
294    }
