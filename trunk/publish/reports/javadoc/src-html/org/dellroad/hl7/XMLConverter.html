035/**
036 * Converts {@link HL7Message}s to and from XML.
037 */
038public final class XMLConverter {
039
040    /**
041     * HL7 message XML namespace.
042     */
043    public static final String HL7_NAMESPACE_URI = null;
044
045    /**
046     * XML tag name used by {@link #toXML}.
047     */
048    public static final String HL7_TAG = "HL7";
049
050    /**
051     * XML tag name used for one XML-encoded message.
052     */
053    public static final String MESSAGE_TAG = "MESSAGE";
054
055    private XMLConverter() {
056    }
057
058    /**
059     * Create an empty XML document for HL7 messages.
060     */
061    public static Document createDocument() {
062        DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
063        documentBuilderFactory.setNamespaceAware(true);
064        DocumentBuilder documentBuilder;
065        try {
066            documentBuilder = documentBuilderFactory.newDocumentBuilder();
067        } catch (ParserConfigurationException e) {
068            throw new RuntimeException(e);
069        }
070        Document document = documentBuilder.getDOMImplementation().createDocument(HL7_NAMESPACE_URI, HL7_TAG, null);
071        document.setXmlStandalone(true);
072        return document;
073    }
074
075    /**
076     * Convert an HL7 message message to an XML document. The document
077     * element will be &lt;HL7&gt; and contain one child which is the
078     * XML encoding of the message.
079     *
080     * @param message HL7 message
081     * @param omitEmpty Omit empty tags (other than the last one)
082     * @return HL7 message encoded as an XML document
083     */
084    public static Document toXML(HL7Message message, boolean omitEmpty) {
085        Document doc = createDocument();
086        XMLConverter.appendMessage(doc.getDocumentElement(), message, omitEmpty);
087        return doc;
088    }
089
090    /**
091     * Convert a message to XML and append it to the given element.
092     *
093     * @param parent an XML {@link Element} or {@link Document} node
094     * @param message HL7 message to append
095     * @param omitEmpty Omit empty tags (other than the last one)
096     */
097    public static void appendMessage(Node parent, HL7Message message, boolean omitEmpty) {
098        Element messageTag = parent.getOwnerDocument().createElementNS(HL7_NAMESPACE_URI, MESSAGE_TAG);
099        for (HL7Segment segment : message.getSegments())
100            XMLConverter.appendSegment(messageTag, segment, omitEmpty);
101        parent.appendChild(messageTag);
102    }
103
104    /**
105     * Convert a message to XML and append it to the given {@link XMLStreamWriter}.
106     *
107     * @param writer XML output
108     * @param message HL7 message to append
109     * @param omitEmpty Omit empty tags (other than the last one)
110     */
111    public static void appendMessage(XMLStreamWriter writer, HL7Message message, boolean omitEmpty) throws XMLStreamException {
112        writer.writeStartElement(MESSAGE_TAG);
113        writer.setDefaultNamespace(HL7_NAMESPACE_URI);
114        for (HL7Segment segment : message.getSegments())
115            XMLConverter.appendSegment(writer, segment, omitEmpty);
116        writer.writeEndElement();
117    }
118
119    /**
120     * Convert a segment to XML and append it to the given element.
121     *
122     * @param parent an XML {@link Element} or {@link Document} node
123     * @param segment HL7 segment to append
124     * @param omitEmpty Omit empty tags (other than the last one)
125     */
126    public static void appendSegment(Node parent, HL7Segment segment, boolean omitEmpty) {
127        String segName = segment.getName();
128        Document doc = parent.getOwnerDocument();
129        Element segXML = doc.createElementNS(HL7_NAMESPACE_URI, segName);
130        HL7Field[] fields = segment.getFields();
131        for (int i = 1; i < fields.length; i++) {
132            HL7Field field = fields[i];
133            if (omitEmpty && i < fields.length - 1 && field.isEmpty())
134                continue;
135            String fieldTag = segName + "." + i;
136            for (String[][] repeat : field.getValue()) {
137                Element repeatXML = doc.createElementNS(HL7_NAMESPACE_URI, fieldTag);
138                segXML.appendChild(repeatXML);
139                if (repeat.length == 1 && repeat[0].length == 1) {
140                    if (repeat[0][0].length() > 0)
141                        repeatXML.appendChild(doc.createTextNode(repeat[0][0]));
142                    continue;
143                }
144                for (int j = 0; j < repeat.length; j++) {
145                    String[] comp = repeat[j];
146                    if (omitEmpty
147                      && j < repeat.length - 1
148                      && comp.length == 1 && comp[0].length() == 0)
149                        continue;
150                    String compTag = fieldTag + "." + (j + 1);
151                    Element compXML = doc.createElementNS(HL7_NAMESPACE_URI, compTag);
152                    repeatXML.appendChild(compXML);
153                    if (comp.length == 1) {
154                        if (comp[0].length() > 0)
155                            compXML.appendChild(doc.createTextNode(comp[0]));
156                        continue;
157                    }
158                    for (int k = 0; k < comp.length; k++) {
159                        String subcomp = comp[k];
160                        if (omitEmpty
161                          && k < comp.length - 1 && subcomp.length() == 0)
162                            continue;
163                        String subcompTag = compTag + "." + (k + 1);
164                        Element subcompXML = doc.createElementNS(HL7_NAMESPACE_URI, subcompTag);
165                        if (subcomp.length() > 0)
166                            subcompXML.appendChild(doc.createTextNode(subcomp));
167                        compXML.appendChild(subcompXML);
168                    }
169                }
170            }
171        }
172        parent.appendChild(segXML);
173    }
174
175    /**
176     * Convert a segment to XML and write it to the given {@link XMLStreamWriter}.
177     *
178     * @param writer XML output
179     * @param segment HL7 segment to append
180     * @param omitEmpty Omit empty tags (other than the last one)
181     */
182    public static void appendSegment(XMLStreamWriter writer, HL7Segment segment, boolean omitEmpty) throws XMLStreamException {
183        final String segName = segment.getName();
184        final HL7Field[] fields = segment.getFields();
185        writer.writeStartElement(segName);
186        writer.setDefaultNamespace(HL7_NAMESPACE_URI);
187        for (int i = 1; i < fields.length; i++) {
188            final HL7Field field = fields[i];
189            if (omitEmpty && i < fields.length - 1 && field.isEmpty())
190                continue;
191            final String fieldTag = segName + "." + i;
192            for (String[][] repeat : field.getValue()) {
193                writer.writeStartElement(fieldTag);
194                if (repeat.length == 1 && repeat[0].length == 1) {
195                    if (repeat[0][0].length() > 0)
196                        writer.writeCharacters(repeat[0][0]);
197                    writer.writeEndElement();
198                    continue;
199                }
200                for (int j = 0; j < repeat.length; j++) {
201                    final String[] comp = repeat[j];
202                    if (omitEmpty
203                      && j < repeat.length - 1
204                      && comp.length == 1 && comp[0].length() == 0)
205                        continue;
206                    final String compTag = fieldTag + "." + (j + 1);
207                    writer.writeStartElement(compTag);
208                    if (comp.length == 1) {
209                        if (comp[0].length() > 0)
210                            writer.writeCharacters(comp[0]);
211                        writer.writeEndElement();
212                        continue;
213                    }
214                    for (int k = 0; k < comp.length; k++) {
215                        final String subcomp = comp[k];
216                        if (omitEmpty
217                          && k < comp.length - 1 && subcomp.length() == 0)
218                            continue;
219                        final String subcompTag = compTag + "." + (k + 1);
220                        writer.writeStartElement(subcompTag);
221                        if (subcomp.length() > 0)
222                            writer.writeCharacters(subcomp);
223                        writer.writeEndElement();
224                    }
225                    writer.writeEndElement();
226                }
227                writer.writeEndElement();
228            }
229        }
230        writer.writeEndElement();
231    }

233    /**
234     * Serialize the XML document and write it to the given output.
235     */
236    public static void stream(Document doc, OutputStream out) throws IOException {
237
238        // Create and configure Transformer
239        TransformerFactory transformFactory = TransformerFactory.newInstance();
240        try {
241            transformFactory.setAttribute("indent-number", 2);
242        } catch (IllegalArgumentException e) {
243            // ignore
244        }
245        Transformer transformer;
246        try {
247            transformer = transformFactory.newTransformer();
248        } catch (TransformerConfigurationException e) {
249            throw new RuntimeException(e);
250        }
251        transformer.setOutputProperty(OutputKeys.METHOD, "xml");
252        transformer.setOutputProperty(OutputKeys.INDENT, "yes");
253
254        // Transform DOM into serialized output stream.
255        // Wrap output stream in a writer to work around this bug:
256        // http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6337981
257        OutputStreamWriter writer = new OutputStreamWriter(out, "UTF-8");
258        try {
259            transformer.transform(new DOMSource(doc), new StreamResult(writer));
260        } catch (TransformerException e) {
261            if (e.getCause() instanceof IOException)
262                throw (IOException)e.getCause();
263            throw new RuntimeException(e);
264        }
265        writer.flush();
266    }

268    /**
269     * Test routine. Reads HL7 messages in "file format" and outputs them in an XML document.
270     */
271    @SuppressWarnings("fallthrough")
272    public static void main(String[] args) throws Exception {
273        InputStream in = System.in;
274        boolean verbose = false;
275        if (args.length > 0 && args[0].equals("-v")) {
276            verbose = true;
277            String[] args2 = new String[args.length - 1];
278            System.arraycopy(args, 1, args2, 0, args2.length);
279            args = args2;
280        }
281        switch (args.length) {
282        case 1:
283            in = new FileInputStream(args[0]);
284            // FALLTHROUGH
285        case 0:
286            break;
287        default:
288            System.err.println("Usage: XMLConverter [-v] [filename]");
289            System.exit(1);
290            break;
291        }
292        HL7FileReader reader = new HL7FileReader(in);
293        Document doc = createDocument();
294        while (true) {
295            try {
296                XMLConverter.appendMessage(doc.getDocumentElement(), reader.readMessage(), !verbose);
297            } catch (EOFException e) {
298                break;
299            }
300        }
301        reader.close();
302        XMLConverter.stream(doc, System.out);
303    }
