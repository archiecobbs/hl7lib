012/**
013 * Container for the HL7 separator and escape characters for an HL7 message.
014 */
015@SuppressWarnings("serial")
016public final class HL7Seps implements Serializable {
017
018    public static final char DEFAULT_FIELD_SEPARATOR = '|';
019    public static final char DEFAULT_COMPONENT_SEPARATOR = '^';
020    public static final char DEFAULT_REPEAT_SEPARATOR = '~';
021    public static final char DEFAULT_SUBCOMPONENT_SEPARATOR = '&';
022    public static final char DEFAULT_ESCAPE_CHARACTER = '\\';
023
024    public static final char FIELD_SEPARATOR_ESCAPE = 'F';
025    public static final char COMPONENT_SEPARATOR_ESCAPE = 'S';
026    public static final char REPEAT_SEPARATOR_ESCAPE = 'R';
027    public static final char ESCAPE_CHARACTER_ESCAPE = 'E';
028    public static final char SUBCOMPONENT_SEPARATOR_ESCAPE = 'T';
029
030    /**
031     * Separator using the default HL7 separator and escape characters.
032     */
033    public static final HL7Seps DEFAULT;
034    static {
035        try {
036            DEFAULT = new HL7Seps(DEFAULT_FIELD_SEPARATOR, DEFAULT_COMPONENT_SEPARATOR,
037              DEFAULT_REPEAT_SEPARATOR, DEFAULT_ESCAPE_CHARACTER, DEFAULT_SUBCOMPONENT_SEPARATOR);
038        } catch (HL7ContentException e) {
039            throw new RuntimeException("impossible", e);
040        }
041    }

043    private final char fieldSep;
044    private final char compSep;
045    private final char repSep;
046    private final char escChar;
047    private final char subSep;

049    /**
050     * Constructor.
051     *
052     * <p>
053     * The sub-component separator and escape characters are optional and may be equal to <code>(char)0</code>
054     * to indicate not defined. However, if a sub-component separator is defined, then so must be an escape
055     * character.
056     *
057     * @param fieldSep field separator
058     * @param compSep component separator
059     * @param repSep repetition separator
060     * @param escChar escape character, or <code>(char)0</code> for none
061     * @param subSep subcomponent separator, or <code>(char)0</code> for none
062     * @throws HL7ContentException if the characters are invalid according to {@link #validate}
063     */
064    public HL7Seps(char fieldSep, char compSep, char repSep, char escChar, char subSep) throws HL7ContentException {
065        validate(fieldSep, compSep, repSep, escChar, subSep);
066        this.fieldSep = fieldSep;
067        this.compSep = compSep;
068        this.repSep = repSep;
069        this.escChar = escChar;
070        this.subSep = subSep;
071    }

073    /**
074     * Convenience constructor for when there is no sub-component separator.
075     * Equivalent to:
076     *  <blockquote>
077     *  <code>HL7Seps(fieldSep, compSep, repSep, escChar, (char)0)</code>
078     *  </blockquote>
079     *
080     * @param fieldSep field separator
081     * @param compSep component separator
082     * @param repSep repetition separator
083     * @param escChar escape character, or <code>(char)0</code> for none
084     * @throws HL7ContentException if the characters are invalid according to {@link #validate}
085     */
086    public HL7Seps(char fieldSep, char compSep, char repSep, char escChar) throws HL7ContentException {
087        this(fieldSep, compSep, repSep, escChar, '\u0000');
088    }

091     * Convenience constructor for when there is no sub-component separator and no escape character.
092     * Equivalent to:
093     *  <blockquote>
094     *  <code>HL7Seps(fieldSep, compSep, repSep, (char)0, (char)0)</code>
095     *  </blockquote>
096     *
097     * @param fieldSep field separator
098     * @param compSep component separator
099     * @param repSep repetition separator
100     * @throws HL7ContentException if the characters are invalid according to {@link #validate}
101     */
102    public HL7Seps(char fieldSep, char compSep, char repSep) throws HL7ContentException {
103        this(fieldSep, compSep, repSep, '\u0000', '\u0000');
104    }

153    /**
154     * Escape instances of any separator or escape character within the given string.
155     *
156     * <p>
157     * If some character needs to be escaped but there is no escape character defined, the character is silently elided.
158     *
159     * @param value the String to escape
160     * @return the escaped string
161     */
162    public String escape(String value) {
163        StringBuilder buf = new StringBuilder(value.length());
164        this.escape(value, buf);
165        return buf.toString();
166    }
167
168    /**
169     * Escape instances of any separator or escape character within the given string, and add the result to the given buffer.
170     *
171     * <p>
172     * If some character needs to be escaped but there is no escape character defined, the character is silently elided.
173     *
174     * @param value the String to escape
175     * @param buf buffer to append to
176     */
177    public void escape(String value, StringBuilder buf) {
178        final int length = value.length();
179        for (int i = 0; i < length; i++) {
180            char ch = value.charAt(i);
181            char escapeCode;
182            if (ch == this.fieldSep)
183                escapeCode = FIELD_SEPARATOR_ESCAPE;
184            else if (ch == this.repSep)
185                escapeCode = REPEAT_SEPARATOR_ESCAPE;
186            else if (ch == this.compSep)
187                escapeCode = COMPONENT_SEPARATOR_ESCAPE;
188            else if (ch == this.subSep && this.subSep != '\u0000')
189                escapeCode = SUBCOMPONENT_SEPARATOR_ESCAPE;
190            else if (ch == this.escChar && this.escChar != '\u0000')
191                escapeCode = ESCAPE_CHARACTER_ESCAPE;
192            else {
193                buf.append(ch);
194                continue;
195            }
196            if (this.escChar != '\u0000') {
197                buf.append(this.escChar);
198                buf.append(escapeCode);
199                buf.append(this.escChar);
200            }
201        }
202    }

204    /**
205     * Determine if there is a sub-component character defined.
206     */

208        return this.subSep != '\u0000';
209    }

213     */

215        return this.escChar != '\u0000';
216    }

218    /**
219     * Parse the escaped string back into its unescaped form.
220     *
221     * <p>
222     * The string must have been previously escaped using the same escape character as is defined for this instance.
223     *
224     * <p>
225     * If no escape character is defined for this instance, the unaltered value is returned.
226     *
227     * <p>
228     * Any "custom" or invalid escapes are silently removed.
229     *
230     * <p>
231     * An unclosed escape is returned unaltered.
232     *
233     * @param value escaped string value
234     * @return original unescaped value
235     */
236    public String unescape(String value) {
237
238        // Optimize for the common case
239        if (this.escChar == '\u0000' || value.indexOf(this.escChar) == -1)
240            return value;
241
242        // Parse out escapes
243        StringBuilder buf = new StringBuilder(value.length());
244        int[] escapes = HL7Util.find(value, this.escChar);
245        int posn = 0;
246        for (int i = 0; i < escapes.length; i++) {
247
248            // Append substring prior to escape
249            buf.append(value.substring(posn, escapes[i]));
250
251            // No more escapes? We're done
252            if (++i == escapes.length)
253                break;
254
255            // Check for unclosed escape (ignore it)
256            if (i == escapes.length - 1) {
257                buf.append(value.substring(escapes[i - 1]));
258                break;
259            }
260
261            // Check for a known escape match
262            if (escapes[i] == escapes[i - 1] + 2) {
263                switch (value.charAt(escapes[i] - 1)) {
264                case FIELD_SEPARATOR_ESCAPE:
265                    buf.append(this.fieldSep);
266                    break;
267                case COMPONENT_SEPARATOR_ESCAPE:
268                    buf.append(this.compSep);
269                    break;
270                case REPEAT_SEPARATOR_ESCAPE:
271                    buf.append(this.repSep);
272                    break;
273                case ESCAPE_CHARACTER_ESCAPE:
274                    buf.append(this.escChar);   // we know escChar != '\u0000'
275                    break;
276                case SUBCOMPONENT_SEPARATOR_ESCAPE:
277                    if (this.subSep != '\u0000') {
278                        buf.append(this.subSep);
279                        break;
280                    }
281                    // Elide unknown escape
282                    break;
283                default:
284                    // Elide unknown escape
285                    break;
286                }
287            }
288
289            // Update next starting position
290            posn = escapes[i] + 1;
291        }
292
293        // Done
294        return buf.toString();
295    }
296
297    /**
298     * Returns string representation as well-formed MSH.1 and MSH.2 fields.
299     */
300    @Override
301    public String toString() {
302        StringBuilder buf = new StringBuilder(5);
303        buf.append(this.fieldSep);
304        buf.append(this.compSep);
305        buf.append(this.repSep);
306        if (this.escChar != '\u0000') {
307            buf.append(this.escChar);
308            if (this.subSep != '\u0000')
309                buf.append(this.subSep);
310        }
311        return buf.toString();
312    }

314    @Override
315    public boolean equals(Object obj) {
316        if (obj == null || obj.getClass() != getClass())
317            return false;
318        HL7Seps that = (HL7Seps)obj;
319        return this.fieldSep == fieldSep
320          && this.compSep == compSep
321          && this.repSep == repSep
322          && this.escChar == escChar
323          && this.subSep == subSep;
324    }

335    /**
336     * Verify that the given combination of separator and escape characters is valid.
337     *
338     * <p>
339     * This method performs the following checks:
340     * <ul>
341     *  <li>All defined characters are plain ASCII.</li>
342     *  <li>All defined characters are unique.</li>
343     *  <li>If a sub-component separator is defined, then so is an escape character.</li>
344     * </ul>
345     *
346     * @param fieldSep field separator
347     * @param compSep component separator
348     * @param repSep repetition separator
349     * @param escChar escape character, or <code>(char)0</code> for none
350     * @param subSep subcomponent separator, or <code>(char)0</code> for none
351     * @throws HL7ContentException if the characters are invalid
352     */
353    public static void validate(char fieldSep, char compSep, char repSep, char escChar, char subSep) throws HL7ContentException {
354
355        // Line them up
356        char[] chars = new char[] { fieldSep, compSep, repSep, escChar, subSep };
357        String[] names = new String[] {
358          "field separator",
359          "component separator",
360          "repeat separator",
361          "escape character",
362          "sub-component separator"
363        };
364
365        // Check for character validity
366        for (int i = 0; i < chars.length; i++) {
367            if (i >= 3 && chars[i] == '\u0000')
368                continue;
369            if (chars[i] <= ' ' || chars[i] > '~')
370                throw new HL7ContentException("illegal " + names[i] + " `" + chars[i] + "'");
371        }
372
373        // Sub-component defined implies escape defined
374        if (escChar == '\u0000' && subSep != '\u0000')
375            throw new HL7ContentException("escape character must be defined when sub-component separator is defined");
376
377        // Check for duplicates
378        for (int i = 0; i < chars.length - 1; i++) {
379            if (i >= 3 && chars[i] == '\u0000')
380                continue;
381            for (int j = i + 1; j < chars.length; j++) {
382                if (chars[i] == chars[j])
383                    throw new HL7ContentException("duplicate " + names[i] + " and " + names[j] + " `" + chars[i] + "'");
384            }
385        }
386    }
