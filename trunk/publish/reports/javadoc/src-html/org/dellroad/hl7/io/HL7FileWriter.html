008package org.dellroad.hl7.io;
009
010import java.io.BufferedWriter;
011import java.io.IOException;
012import java.io.OutputStream;
013import java.io.OutputStreamWriter;
014import java.io.Writer;
015
016import org.dellroad.hl7.HL7Message;
017import org.dellroad.hl7.HL7Segment;
018import org.dellroad.hl7.HL7Seps;
019import org.dellroad.hl7.HL7Writer;
020
021/**
022 * Writes out HL7 messages in "file format".
023 *
024 * <p>
025 * Each segment is terminated with LF, and a blank line is added between messages.
026 */
027public class HL7FileWriter implements HL7Writer {
028
029    protected final BufferedWriter writer;
030
031    /**
032     * Constructor.
033     *
034     * @param out underlying writer
035     */
036    public HL7FileWriter(Writer out) {
037        if (out == null)
038            throw new IllegalArgumentException("null out");
039        this.writer = new BufferedWriter(out);
040    }
041
042    /**
043     * Convenience constructor. Equivalent to:
044     *  <blockquote>
045     *  <code>HL7FileWriter(new OutputStreamWriter(out))</code>
046     *  </blockquote>
047     *
048     * @param out underlying input stream
049     */
050    public HL7FileWriter(OutputStream out) {
051        this(new OutputStreamWriter(out));
052    }
053
054    /**
055     * Write the given message to the underlying output (and then flush it).
056     */
057    public void writeMessage(HL7Message message) throws IOException {
058        HL7Seps seps = message.getMSHSegment().getHL7Seps();
059        for (HL7Segment segment : message.getSegments()) {
060            this.writer.write(segment.toString(seps));
061            this.writer.write('\n');
062        }
063        this.writer.write('\n');
064        this.writer.flush();
065    }
066
067    /**
068     * Closes the underlying writer.
069     */
070    public void close() throws IOException {
071        this.writer.close();
072    }
073}
