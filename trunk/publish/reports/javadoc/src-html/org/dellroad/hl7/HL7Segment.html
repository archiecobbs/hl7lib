008    package org.dellroad.hl7;
009    
010    import java.io.Serializable;
011    import java.util.ArrayList;
012    
013    /**
014     * Represents one segment in an HL7 message.
015     */
016    @SuppressWarnings("serial")
017    public class HL7Segment implements Serializable {
018    
019        protected final ArrayList<HL7Field> fields = new ArrayList<HL7Field>();
020    
021        /**
022         * Constructor for {@link MSHSegment} use only.
023         */
024        HL7Segment() {
025        }
026    
027        /**
028         * Constructor that initializes the name of the segment only.
029         *
030         * @param name segment name
031         * @throws HL7ContentException if the name is an invalid segment name
032         */
033        public HL7Segment(String name) throws HL7ContentException {
034            this.setName(name);
035        }
036    
037        /**
038         * Constructor taking segment name and explicit fields.
039         *
040         * @param name segment name
041         * @param fields zero or more segment fields
042         * @throws HL7ContentException if the name is an invalid segment name
043         * @throws IllegalArgumentException if any field in the array is null
044         */
045        public HL7Segment(String name, HL7Field[] fields) throws HL7ContentException {
046            this(name);
047            for (HL7Field field : fields) {
048                if (field == null)
049                    throw new IllegalArgumentException("null field");
050                this.fields.add(field);
051            }
052        }
053    
054        /**
055         * Constructor that parses a segment line.
056         *
057         * @param line segment line
058         * @param seps separator and escape characters
059         * @throws HL7ContentException if the segment name is invalid
060         */
061        public HL7Segment(String line, HL7Seps seps) throws HL7ContentException {
062            this.parseAndAddFields(line, seps);
063        }
064    
065        /**
066         * Copy constructor.
067         */
068        public HL7Segment(HL7Segment segment) {
069            for (HL7Field field : segment.getFields())
070                this.appendField(new HL7Field(field));
071        }
072    
073        /**
074         * Get segment name, e.g., <code>MSH</code>, <code>PV1</code>, etc.
075         * Equivalent to:
076         * <blockquote>
077         * <code>getField(0).get(0, 0, 0)</code>
078         * </blockquote>
079         */
080        public String getName() {
081            return this.fields.get(0).get(0, 0, 0);
082        }
083    
084        /**
085         * Set segment name.
086         *
087         * @param name new segment name
088         * @throws HL7ContentException if the name is an invalid segment name
089         */
090        public void setName(String name) throws HL7ContentException {
091            this.checkSegmentName(name);
092            HL7Field field = new HL7Field(name);
093            if (this.fields.isEmpty())
094                this.fields.add(field);
095            else
096                this.fields.set(0, field);
097        }
098    
099        /**
100         * Get number of fields (including segment name).
101         */
102        public int getNumFields() {
103            return this.fields.size();
104        }
105    
106        /**
107         * Add a field to the end of this segment.
108         *
109         * @param field field to add
110         * @throws IllegalArgumentException if field is null
111         */
112        public void appendField(HL7Field field) {
113            if (field == null)
114                throw new IllegalArgumentException("field is null");
115            this.fields.add(field);
116        }
117    
118        /**
119         * Set a field in this segment. If the segment doesn't have <code>index</code> fields, intermediate empty fields will
120         * be added.
121         *
122         * <p>
123         * This method cannot be used to set the segment name. Use {@link #setName} for that.
124         *
125         * @param field field to set
126         * @param index index of the field in the segment (where zero is the segment name); must be at least one
127         * @throws IllegalArgumentException if field is null
128         * @throws IllegalArgumentException if index is zero or less
129         */
130        public void setField(int index, HL7Field field) {
131            if (field == null)
132                throw new IllegalArgumentException("field is null");
133            if (index < 1)
134                throw new IllegalArgumentException("index=" + index);
135            while (this.fields.size() < index + 1)
136                this.fields.add(HL7Field.EMPTY);
137            this.fields.set(index, field);
138        }
139    
140        /**
141         * Convenience method. Equivalent to:
142         * <blockquote>
143         * <code>setField(index, new HL7Field(value))</code>
144         * </blockquote>
145         *
146         * @param value
147         */
148        public void setField(int index, String value) {
149            this.setField(index, new HL7Field(value));
150        }
151    
152        /**
153         * Get a field by "HL7 index". E.g., if this message is "ABC", an index of 3 would return ABC.3. If index is zero, the name
154         * of this segment is returned.
155         *
156         * @param index of the requested field
157         * @return the requested field, or null if not present in this segment
158         * @throws IllegalArgumentException if <code>index</code> is negative
159         */
160        public HL7Field getField(int index) {
161            if (index < 0)
162                throw new IllegalArgumentException("index=" + index);
163            if (index >= this.fields.size())
164                return null;
165            return this.fields.get(index);
166        }
167    
168        /**
169         * Get all of the fields, starting with the segment name.
170         */
171        public HL7Field[] getFields() {
172            return this.fields.toArray(new HL7Field[this.fields.size()]);
173        }
174    
175        /**
176         * Trim off all but the first <code>size</code> fields. If the current number of fields is already
177         * less than or equal to <code>size</code> then nothing happens.
178         *
179         * @throws IllegalArgumentException if <code>size</code> is less than one
180         */
181        public void trimTo(int size) {
182            if (size < 1)
183                throw new IllegalArgumentException("size < 1");
184            while (size < this.fields.size())
185                this.fields.remove(this.fields.size() - 1);
186            this.fields.trimToSize();
187        }
188    
189        /**
190         * Append string encoding of this segment to the provided buffer.
191         *
192         * @param seps HL7 separator and escape characters
193         */
194        public void append(StringBuilder buf, HL7Seps seps) {
195            boolean first = true;
196            for (HL7Field field : this.fields) {
197                if (first)
198                    first = false;
199                else
200                    buf.append(seps.getFieldSep());
201                field.append(buf, seps);
202            }
203        }
204    
205        /**
206         * Convert to a string using the provided separators.
207         *
208         * @param seps HL7 separator and escape characters
209         */
210        public String toString(HL7Seps seps) {
211            StringBuilder buf = new StringBuilder();
212            this.append(buf, seps);
213            return buf.toString();
214        }
215    
216        /**
217         * Convert to a string using {@link HL7Seps#DEFAULT}.
218         */
219        @Override
220        public String toString() {
221            return this.toString(HL7Seps.DEFAULT);
222        }
223    
224        @Override
225        public boolean equals(Object obj) {
226            if (obj == null || obj.getClass() != getClass())
227                return false;
228            HL7Segment that = (HL7Segment)obj;
229            return this.fields.equals(that.fields);
230        }
231    
232        @Override
233        public int hashCode() {
234            return this.fields.hashCode();
235        }
236    
237        /**
238         * Parse fields from given line and add them.
239         *
240         * @param line segment or segment fragment
241         * @param seps separator and escape characters
242         * @throws HL7ContentException if this segment contains zero fields and the first field is not a valid segment name
243         */
244        protected void parseAndAddFields(String line, HL7Seps seps) throws HL7ContentException {
245            int[] fieldSeps = HL7Util.find(line, seps.getFieldSep());
246            int posn = 0;
247            for (int i = 0; i < fieldSeps.length; i++) {
248                String string = line.substring(posn, fieldSeps[i]);
249                if (this.fields.isEmpty())
250                    this.checkSegmentName(string);
251                this.fields.add(new HL7Field(string, seps));
252                posn = fieldSeps[i] + 1;
253            }
254        }
255    
256        /**
257         * Verify segment name is legit.
258         */
259        private void checkSegmentName(String name) throws HL7ContentException {
260            if (name.length() != 3)
261                throw new HL7ContentException("invalid segment name `" + name + "'");
262        }
263    }
