013    /**
014     * Represents an MSH segment.
015     *
016     * <p>
017     * This subclass does not allow accessing MSH.1 or MSH.2 except
018     * indirectly via {@link #getHL7Seps} and {@link #setHL7Seps}.
019     */
020    @SuppressWarnings("serial")
021    public class MSHSegment extends HL7Segment {
022    
023        public static final String DEFAULT_PROCESSING_ID = "P";
024        public static final String DEFAULT_VERSION_ID = "2.3";
025    
026        public static final String MSH_SEGMENT_NAME = "MSH";
027        public static final String MSA_SEGMENT_NAME = "MSA";
028    
029        public static final String TIMESTAMP_FORMAT = "yyyyMMddHHmmss.SSSZ";
030    
031        private static final HL7Field ACK_MSH_9 = new HL7Field("ACK");
032        private static final HL7Field ACK_MSA_1 = new HL7Field("AA");
033    
034        /**
035         * Parsing constructor.
036         *
037         * @param line encoded HL7 MSH segment
038         * @throws HL7ContentException if segment is invalid
039         */
040        public MSHSegment(String line) throws HL7ContentException {
041    
042            // Parse "MSH", MSH.1, and MSH.2
043            if (!line.startsWith(MSH_SEGMENT_NAME))
044                throw new HL7ContentException("MSH segment does not start with `" + MSH_SEGMENT_NAME + "'");
045            this.fields.add(new HL7Field(MSH_SEGMENT_NAME));
046            if (line.length() < 6)
047                throw new HL7ContentException("MSH segment is truncated");
048            char fieldSep = line.charAt(3);
049            char repSep = line.charAt(5);
050            char compSep = line.charAt(4);
051            char subSep = '\u0000';
052            char escChar = '\u0000';
053            int fieldSep23 = 6;
054            if (line.length() >= 7 && line.charAt(6) != fieldSep) {
055                fieldSep23++;
056                escChar = line.charAt(6);
057                if (line.length() >= 8 && line.charAt(7) != fieldSep) {
058                    fieldSep23++;
059                    subSep = line.charAt(7);
060                }
061            }
062            HL7Seps seps = new HL7Seps(fieldSep, compSep, repSep, escChar, subSep);
063            this.setHL7Seps(seps);
064    
065            // Parse remaining fields (if any)
066            if (fieldSep23 == line.length())
067                return;
068            if (line.charAt(fieldSep23) != fieldSep)
069                throw new HL7ContentException("bogus extra characters in MSH.2");
070            this.parseAndAddFields(line.substring(fieldSep23 + 1), seps);
071        }

073        /**
074         * Constructor.
075         *
076         * This constructor initializes only MSH.1 and MSH.2.
077         *
078         * @param seps separator and escape characters for this message
079         */
080        public MSHSegment(HL7Seps seps) {
081            this.fields.add(new HL7Field(MSH_SEGMENT_NAME));
082            this.setHL7Seps(seps);
083        }

210        /**
211         * Get the separator and escape characters defined by MSH.1 and MSH.2.
212         */
213        public HL7Seps getHL7Seps() {
214            String msh1 = this.getField(1).get(0, 0, 0);
215            String msh2 = this.getField(2).get(0, 0, 0);
216            try {
217                return new HL7Seps(msh1.charAt(0), msh2.charAt(0), msh2.charAt(1),
218                  msh2.length() > 2 ? msh2.charAt(2) : '\u0000',
219                  msh2.length() > 3 ? msh2.charAt(3) : '\u0000');
220            } catch (HL7ContentException e) {
221                throw new RuntimeException("impossible", e);
222            }
223        }
224    
225        /**
226         * Set MSH.1 and MSH.2 based on the provided separator and
227         * escape characters.
228         */
229        public void setHL7Seps(HL7Seps seps) {
230            super.setField(1, new HL7Field("" + seps.getFieldSep()));
231            super.setField(2, new HL7Field("" + seps.getCompSep() + seps.getRepSep()
232              + (seps.hasEscapeCharacter() ? "" + seps.getEscChar() : "")
233              + (seps.hasSubcomponentSeparator() ? "" + seps.getSubSep() : "")));
234        }

261        /**
262         * Create an acknowlegement message for this message.
263         *
264         * @param serno next local HL7 serial number
265         * @return an acknowlegement of this message
266         * @throws HL7ContentException if the required fields are not present
267         *  in this instance
268         */
269        public HL7Message createACK(long serno) throws HL7ContentException {
270    
271            // We must have at least up to MSH.11
272            if (getNumFields() < 12)
273                throw new HL7ContentException("insufficient fields for ACK'ing");
274            HL7Field versionID = this.getVersionID() != null ? this.getVersionID() : new HL7Field(DEFAULT_VERSION_ID);
275    
276            // Build ACK
277            HL7Message ack = new HL7Message();
278            MSHSegment msh = ack.getMSHSegment();
279            msh.setTimestamp(new HL7Field(new SimpleDateFormat(TIMESTAMP_FORMAT).format(new Date())));
280            msh.setMessageType(ACK_MSH_9);
281            msh.setControlID(new HL7Field("" + serno));
282            msh.setProcessingID(this.getProcessingID());
283            msh.setVersionID(versionID);
284            HL7Segment msa = new HL7Segment(MSA_SEGMENT_NAME);
285            msa.setField(1, ACK_MSA_1);
286            msa.setField(2, this.getControlID());
287            ack.getSegments().add(msa);
288    
289            // Done
290            return ack;
291        }
292    
293        /**
294         * Check whether the given message is an acknowlegement of this MSH.
295         *
296         * @param ack putative acknowlegement of this message
297         * @throws HL7ContentException if the required fields are not present
298         *  in this instance
299         */
300        public boolean isACK(HL7Message ack) throws HL7ContentException {
301            HL7Field controlID = this.getControlID();
302            HL7Field procID = this.getProcessingID();
303            if (controlID == null || procID == null)
304                throw new HL7ContentException("insufficient fields for ACK'ing");
305            MSHSegment ackMSH = ack.getMSHSegment();
306            return ACK_MSH_9.equals(ackMSH.getMessageType())
307              && controlID.equals(ack.getField("MSA.2"))
308              && procID.equals(ackMSH.getProcessingID())
309              && ACK_MSA_1.equals(ack.getField("MSA.1"));
310        }

312        /**
313         * Append string encoding of this segment to the provided buffer.
314         *
315         * <p>
316         * This overridden version ensures that MSH.1 and MSH.2 are not escaped.
317         */
318        @Override
319        public void append(StringBuilder buf, HL7Seps seps) {
320            HL7Field[] fields = this.fields.toArray(new HL7Field[this.fields.size()]);
321            buf.append(fields[0]);
322            buf.append(seps);
323            for (int i = 3; i < fields.length; i++) {
324                buf.append(seps.getFieldSep());
325                fields[i].append(buf, seps);
326            }
327        }
328    
329        /**
330         * Convert to a string using the provided separators.
331         *
332         * <p>
333         * This overridden version ensures that MSH.1 and MSH.2 are consistent with <code>seps</code> in the generated string.
334         * This separators configured in this instance are not affected regardless.
335         */
336        @Override
337        public String toString(HL7Seps seps) {
338            if (seps.equals(getHL7Seps()))
339                return super.toString(seps);
340            MSHSegment copy = new MSHSegment(this);
341            copy.setHL7Seps(seps);
342            return copy.toString();
343        }
344    
345        @Override
346        public String toString() {
347            return super.toString(getHL7Seps());
348        }
