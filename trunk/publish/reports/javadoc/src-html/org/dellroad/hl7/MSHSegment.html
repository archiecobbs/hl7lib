013/**
014 * Represents an MSH segment.
015 *
016 * <p>
017 * This subclass does not allow accessing MSH.1 or MSH.2 except
018 * indirectly via {@link #getHL7Seps} and {@link #setHL7Seps setHL7Seps()}.
019 */
020@SuppressWarnings("serial")

022
023    public static final String DEFAULT_PROCESSING_ID = "P";
024    public static final String DEFAULT_VERSION_ID = "2.3";
025
026    public static final String MSH_SEGMENT_NAME = "MSH";
027    public static final String MSA_SEGMENT_NAME = "MSA";
028
029    public static final String TIMESTAMP_FORMAT = "yyyyMMddHHmmss.SSSZ";
030
031    private static final String ACK = "ACK";
032    private static final HL7Field ACK_MSH_9 = new HL7Field(ACK);
033    private static final HL7Field ACK_MSA_1 = new HL7Field("AA");
034
035    /**
036     * Parsing constructor.
037     *
038     * @param line encoded HL7 MSH segment
039     * @throws HL7ContentException if segment is invalid
040     */
041    public MSHSegment(String line) throws HL7ContentException {
042
043        // Parse "MSH", MSH.1, and MSH.2
044        if (!line.startsWith(MSH_SEGMENT_NAME))
045            throw new HL7ContentException("MSH segment does not start with `" + MSH_SEGMENT_NAME + "'").setContent(line);
046        this.fields.add(new HL7Field(MSH_SEGMENT_NAME));
047        if (line.length() < 6)
048            throw new HL7ContentException("MSH segment is truncated").setContent(line);
049        char fieldSep = line.charAt(3);
050        char repSep = line.charAt(5);
051        char compSep = line.charAt(4);
052        char subSep = '\u0000';
053        char escChar = '\u0000';
054        int fieldSep23 = 6;
055        if (line.length() >= 7 && line.charAt(6) != fieldSep) {
056            fieldSep23++;
057            escChar = line.charAt(6);
058            if (line.length() >= 8 && line.charAt(7) != fieldSep) {
059                fieldSep23++;
060                subSep = line.charAt(7);
061            }
062        }
063        HL7Seps seps = new HL7Seps(fieldSep, compSep, repSep, escChar, subSep);
064        this.setHL7Seps(seps);
065
066        // Parse remaining fields (if any)
067        if (fieldSep23 == line.length())
068            return;
069        if (line.charAt(fieldSep23) != fieldSep)
070            throw new HL7ContentException("bogus extra characters in MSH.2").setContent(line);
071        this.parseAndAddFields(line.substring(fieldSep23 + 1), seps);
072    }

081    public MSHSegment(HL7Seps seps) {
082        this.fields.add(new HL7Field(MSH_SEGMENT_NAME));
083        this.setHL7Seps(seps);

211    /**
212     * Get the separator and escape characters defined by MSH.1 and MSH.2.
213     */
214    public HL7Seps getHL7Seps() {
215        String msh1 = this.getField(1).get(0, 0, 0);
216        String msh2 = this.getField(2).get(0, 0, 0);
217        try {
218            return new HL7Seps(msh1.charAt(0), msh2.charAt(0), msh2.charAt(1),
219              msh2.length() > 2 ? msh2.charAt(2) : '\u0000',
220              msh2.length() > 3 ? msh2.charAt(3) : '\u0000');
221        } catch (HL7ContentException e) {
222            throw new RuntimeException("impossible", e);
223        }
224    }

226    /**
227     * Set MSH.1 and MSH.2 based on the provided separator and
228     * escape characters.
229     */
230    public void setHL7Seps(HL7Seps seps) {
231        super.setField(1, new HL7Field("" + seps.getFieldSep()));
232        super.setField(2, new HL7Field("" + seps.getCompSep() + seps.getRepSep()
233          + (seps.hasEscapeCharacter() ? "" + seps.getEscChar() : "")
234          + (seps.hasSubcomponentSeparator() ? "" + seps.getSubSep() : "")));
235    }

262    /**
263     * Create an acknowlegement message for this message.
264     *
265     * @param serno next local HL7 serial number
266     * @return an acknowlegement of this message
267     * @throws HL7ContentException if the required fields are not present
268     *  in this instance
269     */
270    public HL7Message createACK(long serno) throws HL7ContentException {
271
272        // We must have at least up to MSH.11
273        if (getNumFields() < 12)
274            throw new HL7ContentException("insufficient fields for ACK'ing");
275        HL7Field versionID = this.getVersionID() != null ? this.getVersionID() : new HL7Field(DEFAULT_VERSION_ID);
276
277        // Build ACK
278        HL7Message ack = new HL7Message();
279        MSHSegment msh = ack.getMSHSegment();
280        msh.setTimestamp(new HL7Field(new SimpleDateFormat(TIMESTAMP_FORMAT).format(new Date())));
281        msh.setMessageType(ACK_MSH_9);
282        msh.setControlID(new HL7Field("" + serno));
283        msh.setProcessingID(this.getProcessingID());
284        msh.setVersionID(versionID);
285        HL7Segment msa = new HL7Segment(MSA_SEGMENT_NAME);
286        msa.setField(1, ACK_MSA_1);
287        msa.setField(2, this.getControlID());
288        ack.getSegments().add(msa);
289
290        // Done
291        return ack;
292    }

294    /**
295     * Check whether the given message is an acknowlegement of this MSH.
296     *
297     * @param ack putative acknowlegement of this message
298     * @throws HL7ContentException if the required fields are not present
299     *  in this instance
300     */
301    public boolean isACK(HL7Message ack) throws HL7ContentException {
302        HL7Field controlID = this.getControlID();
303        HL7Field procID = this.getProcessingID();
304        if (controlID == null || procID == null)
305            throw new HL7ContentException("insufficient fields for ACK'ing");
306        MSHSegment ackMSH = ack.getMSHSegment();
307        return ACK.equals(ack.getField("MSH.9").get(0, 0, 0))
308          && controlID.equals(ack.getField("MSA.2"))
309          && procID.equals(ackMSH.getProcessingID())
310          && ACK_MSA_1.equals(ack.getField("MSA.1"));
311    }

313    /**
314     * Append string encoding of this segment to the provided buffer.
315     *
316     * <p>
317     * This overridden version ensures that MSH.1 and MSH.2 are not escaped.
318     */
319    @Override
320    public void append(StringBuilder buf, HL7Seps seps) {
321        HL7Field[] fields = this.fields.toArray(new HL7Field[this.fields.size()]);
322        buf.append(fields[0]);
323        buf.append(seps);
324        for (int i = 3; i < fields.length; i++) {
325            buf.append(seps.getFieldSep());
326            fields[i].append(buf, seps);
327        }
328    }

330    /**
331     * Convert to a string using the provided separators.
332     *
333     * <p>
334     * This overridden version ensures that MSH.1 and MSH.2 are consistent with <code>seps</code> in the generated string.
335     * This separators configured in this instance are not affected regardless.
336     */
337    @Override
338    public String toString(HL7Seps seps) {
339        if (seps.equals(getHL7Seps()))
340            return super.toString(seps);
341        MSHSegment copy = new MSHSegment(this);
342        copy.setHL7Seps(seps);
343        return copy.toString();
344    }
345
346    @Override
347    public String toString() {
348        return super.toString(getHL7Seps());
349    }
