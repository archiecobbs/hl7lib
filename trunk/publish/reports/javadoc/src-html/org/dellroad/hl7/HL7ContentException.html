008    package org.dellroad.hl7;
009    
010    /**
011     * Exception thrown to indicate invalid HL7 content was encountered.
012     */
013    @SuppressWarnings("serial")
014    public class HL7ContentException extends Exception {
015    
016        public HL7ContentException() {
017        }
018    
019        public HL7ContentException(String message) {
020            super(message);
021        }
022    
023        public HL7ContentException(String message, Throwable cause) {
024            super(message, cause);
025        }
026    
027        public HL7ContentException(Throwable cause) {
028            super(cause);
029        }
030    }
