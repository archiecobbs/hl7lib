008    package org.dellroad.hl7;
009    
010    import java.util.ArrayList;
011    
012    /**
013     * List of {@link HL7Segment} objects that requires there to always be an initial MSH segment.
014     * In particular, this class enforces the following constraints:
015     * <ul>
016     *  <li>There must be at least one item in the list</li>
017     *  <li>The first item in the list must be an {@link MSHSegment}</li>
018     * </ul>
019     *
020     * Illegal operations result in an {@link IllegalStateException} being thrown.
021     *
022     * <p>
023     * Note we do not enforce the other segments to be anything, not even non-null.
024     */
025    @SuppressWarnings("serial")
026    public final class HL7SegmentList extends ArrayList<HL7Segment> {
027    
028        /**
029         * Constructor.
030         *
031         * @throws IllegalArgumentException if <code>msh</code> is null
032         */
033        public HL7SegmentList(MSHSegment msh) {
034            if (msh == null)
035                throw new IllegalArgumentException("null msh");
036            this.add(msh);
037        }
038    
039        @Override
040        public void clear() {
041            throw new IllegalStateException("list can't be empty");
042        }
043    
044        @Override
045        public HL7Segment remove(int index) {
046            if (index == 0 && !(this.get(0) instanceof MSHSegment))
047                throw new IllegalStateException("can't remove initial MSH");
048            return super.remove(index);
049        }
050    
051        @Override
052        public boolean remove(Object obj) {
053            if (obj != null && obj.equals(get(0)))
054                throw new IllegalStateException("can't remove initial MSH");
055            return super.remove(obj);
056        }
057    
058        @Override
059        protected void removeRange(int fromIndex, int toIndex) {
060            if (fromIndex <= 0 && toIndex > 0 && !(this.get(0) instanceof MSHSegment))
061                throw new IllegalStateException("can't remove initial MSH");
062            super.removeRange(fromIndex, toIndex);
063        }
064    
065        @Override
066        public HL7Segment set(int index, HL7Segment segment) {
067            if (index == 0 && !(segment instanceof MSHSegment))
068                throw new IllegalStateException("can't replace initial MSH segment with non-MSH segment");
069            return super.set(index, segment);
070        }
071    }
