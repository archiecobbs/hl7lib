008package org.dellroad.hl7.llp;
009
010import java.io.BufferedInputStream;
011import java.io.EOFException;
012import java.io.IOException;
013import java.io.InputStream;
014
015import org.dellroad.hl7.HL7ContentException;
016import org.dellroad.hl7.HL7Message;
017import org.dellroad.hl7.HL7Reader;
018
019/**
020 * Reads HL7 messages framed in the "lower layer protocol" (LLP).
021 */
022public class LLPInputStream implements HL7Reader {
023
024    private static final int MIN_BUFLEN = 1024;
025    private static final int MAX_BUFLEN = 16 * 1024;
026
027    private final BufferedInputStream inputStream;
028    private final int maxLength;
029
030    private byte[] buf = new byte[MIN_BUFLEN];
031
032    /**
033     * Constructor.
034     *
035     * @param in underlying input stream
036     * @param maxLength maximum allowed message length
037     * @throws IllegalArgumentException if out is null
038     * @throws IllegalArgumentException if maxLength is negative
039     */
040    public LLPInputStream(InputStream in, int maxLength) {
041        if (in == null)
042            throw new IllegalArgumentException("in is null");
043        if (maxLength < 0)
044            throw new IllegalArgumentException("maxLength is negative");
045        this.inputStream = new BufferedInputStream(in);
046        this.maxLength = maxLength;
047    }
048
049    /**
050     * Read next message from the underlying stream.
051     *
052     * @throws EOFException if there is no more input
053     * @throws HL7ContentException if a malformed message is read
054     * @throws LLPException if illegal framing byte(s) are read from the underlying stream, or the message is too long
055     * @throws IOException if an error occurs on the underlying stream
056     */
057    public HL7Message readMessage() throws IOException, HL7ContentException {
058
059        // Read leading byte
060        this.readByte(LLPConstants.LEADING_BYTE);
061
062        // Read message until first trailing byte
063        int len = 0;
064        while (true) {
065            int ch;
066            if ((ch = this.inputStream.read()) == -1)
067                throw new EOFException();
068            if (ch == LLPConstants.TRAILING_BYTE_0)
069                break;
070            if (len >= this.maxLength)
071                throw new LLPException("message is too long (greater than " + this.maxLength + " bytes)");
072            if (len == this.buf.length) {
073                byte[] newbuf = new byte[this.buf.length * 2];
074                System.arraycopy(this.buf, 0, newbuf, 0, len);
075                this.buf = newbuf;
076            }
077            this.buf[len++] = (byte)ch;
078        }
079
080        // Read second trailing byte
081        this.readByte(LLPConstants.TRAILING_BYTE_1);
082
083        // Extract message text
084        String text = new String(buf, 0, len, LLPConstants.CHARACTER_ENCODING);
085        if (this.buf.length > MAX_BUFLEN)
086            this.buf = new byte[MIN_BUFLEN];
087
088        // Return parsed message
089        try {
090            return new HL7Message(text);
091        } catch (HL7ContentException e) {
092            throw e.setContent(text);
093        }
094    }
095
096    /**
097     * Advance past the end of the current message. This method can be used (for example) to skip over the remaining portion
098     * of a badly framed message that resulted in a {@link LLPException} in an attempt to salvage the connection.
099     *
100     * <p>
101     * This method just reads until the next occurrence of a {@link LLPConstants#TRAILING_BYTE_0} byte
102     * followed immediately by a {@link LLPConstants#TRAILING_BYTE_1} byte, then returns.
103     *
104     * @throws EOFException if there is no more input
105     * @throws IOException if an error occurs on the underlying stream
106     */
107    public void skip() throws IOException {
108        int state = 0;
109        while (true) {
110            int ch = this.inputStream.read();
111            if (ch == -1)
112                throw new EOFException();
113            if (state == 0) {
114                if (ch == LLPConstants.TRAILING_BYTE_0)
115                    state = 1;
116            } else {
117                if (ch == LLPConstants.TRAILING_BYTE_1)
118                    return;
119                state = 0;
120            }
121        }
122    }
123
124    private void readByte(int value) throws IOException {
125        int ch;
126        if ((ch = this.inputStream.read()) == -1)
127            throw new EOFException();
128        if (ch != value) {
129            String expected = String.format("0x%02x", value);
130            String actual = String.format("0x%02x", ch);
131            throw new LLPException("expected to read " + expected + " but read " + actual + " instead");
132        }
133    }
134
135    /**
136     * Close the underlying stream.
137     */
138    public void close() throws IOException {
139        this.inputStream.close();
140    }
141}
