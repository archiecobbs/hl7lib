008    package org.dellroad.hl7;
009    
010    import java.io.IOException;
011    
012    /**
013     * Input source for HL7 messages.
014     */
015    public interface HL7Reader {
016    
017        /**
018         * Read the next HL7 message.
019         *
020         * @throws HL7ContentException if an invalid HL7 message is received, decoded, etc. by this instance
021         * @throws java.io.EOFException if there is no more input
022         * @throws IOException if some other error occurs
023         * @throws IOException if {@link #close} has already been invoked
024         */
025        HL7Message readMessage() throws IOException, HL7ContentException;
026    
027        /**
028         * Close this instance.
029         */
030        void close() throws IOException;
031    }
