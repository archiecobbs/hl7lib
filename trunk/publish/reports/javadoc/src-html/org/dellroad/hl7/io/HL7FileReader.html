008package org.dellroad.hl7.io;
009
010import java.io.BufferedReader;
011import java.io.EOFException;
012import java.io.IOException;
013import java.io.InputStream;
014import java.io.InputStreamReader;
015import java.io.Reader;
016
017import org.dellroad.hl7.HL7ContentException;
018import org.dellroad.hl7.HL7Message;
019import org.dellroad.hl7.HL7Reader;
020import org.dellroad.hl7.HL7Segment;
021import org.dellroad.hl7.HL7Seps;
022import org.dellroad.hl7.MSHSegment;
023
024/**
025 * Reads in HL7 messages in "file format".
026 *
027 * <p>
028 * The "file format" is as follows:
029 *  <ul>
030 *  <li>Each message begins with an {@code MSH} segment</li>
031 *  <li>Each segment is terminated by a CR, LF, or CR-LF</li>
032 *  <li>Blank and whitespace-only lines are ignored</li>
033 *  <li>Lines starting with '#' are ignored</li>
034 *  </ul>
035 */
036public class HL7FileReader implements HL7Reader {
037
038    protected final BufferedReader reader;
039
040    private String nextLine;
041    private boolean closed;
042
043    /**
044     * Constructor.
045     *
046     * @param in underlying reader
047     */
048    public HL7FileReader(Reader in) {
049        if (in == null)
050            throw new IllegalArgumentException("null in");
051        this.reader = new BufferedReader(in);
052    }
053
054    /**
055     * Convenience constructor. Equivalent to:
056     *  <blockquote>
057     *  <code>HL7FileReader(new InputStreamReader(in))</code>
058     *  </blockquote>
059     *
060     * @param in underlying input stream
061     */
062    public HL7FileReader(InputStream in) {
063        this(new InputStreamReader(in));
064    }
065
066    /**
067     * Read next message from the underlying stream.
068     *
069     * @throws EOFException if there is no more input
070     * @throws HL7ContentException if a malformed message is read
071     * @throws IOException if an error occurs on the underlying stream
072     */
073    public HL7Message readMessage() throws IOException, HL7ContentException {
074        String line = nextLine();
075        if (line == null)
076            throw new EOFException();
077        MSHSegment msh = new MSHSegment(line);
078        HL7Seps seps = msh.getHL7Seps();
079        HL7Message message = new HL7Message(msh);
080        while ((line = nextLine()) != null) {
081            if (line.startsWith(MSHSegment.MSH_SEGMENT_NAME)) {
082                this.pushback(line);
083                break;
084            }
085            message.getSegments().add(new HL7Segment(line, seps));
086        }
087        return message;
088    }
089
090    /**
091     * Save the line for next time.
092     *
093     * @throws RuntimeException if there is already one saved.
094     */
095    protected void pushback(String line) {
096        if (this.nextLine != null)
097            throw new RuntimeException("internal error");
098        this.nextLine = line;
099    }
100
101    /**
102     * Read next non-ignorable line.
103     *
104     * @return next relevant line, or null for EOF
105     */
106    protected String nextLine() throws IOException {
107        String line = this.nextLine;
108        this.nextLine = null;
109        if (line != null)
110            return line;
111        if (this.closed)
112            return null;
113        while (true) {
114            line = this.reader.readLine();
115            if (line == null)
116                return null;
117            if (line.length() == 0 || line.charAt(0) == '#')
118                continue;
119            if (line.matches("\\s+"))
120                continue;
121            return line;
122        }
123    }
124
125    /**
126     * Closes the underlying reader.
127     */
128    public void close() throws IOException {
129        this.reader.close();
130        this.nextLine = null;
131        this.closed = true;
132    }
133}
