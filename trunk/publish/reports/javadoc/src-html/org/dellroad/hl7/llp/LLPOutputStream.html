008    package org.dellroad.hl7.llp;
009    
010    import java.io.BufferedOutputStream;
011    import java.io.IOException;
012    import java.io.OutputStream;
013    import java.io.OutputStreamWriter;
014    
015    import org.dellroad.hl7.HL7Message;
016    import org.dellroad.hl7.HL7Segment;
017    import org.dellroad.hl7.HL7Seps;
018    import org.dellroad.hl7.HL7Writer;
019    
020    /**
021     * Writes out HL7 messages using the "lower layer protocol".
022     */
023    public class LLPOutputStream implements HL7Writer {
024    
025        private final BufferedOutputStream outputStream;
026        private final OutputStreamWriter outputWriter;
027    
028        /**
029         * Constructor.
030         *
031         * @param out underlying output stream
032         * @throws IllegalArgumentException if out is null
033         */
034        public LLPOutputStream(OutputStream out) {
035            if (out == null)
036                throw new IllegalArgumentException("out is null");
037            this.outputStream = new BufferedOutputStream(out);
038            this.outputWriter = new OutputStreamWriter(this.outputStream, LLPConstants.CHARACTER_ENCODING);
039        }
040    
041        /**
042         * Write a message using HL7 LLP framing and flush the underlying output.
043         */
044        public void writeMessage(HL7Message message) throws IOException {
045    
046            // Write start byte
047            this.outputStream.write(LLPConstants.LEADING_BYTE);
048    
049            // Write message segments
050            HL7Seps seps = message.getMSHSegment().getHL7Seps();
051            StringBuilder buf = new StringBuilder();
052            for (HL7Segment segment : message.getSegments()) {
053                segment.append(buf, seps);
054                buf.append(HL7Message.SEGMENT_TERMINATOR);
055                this.outputWriter.write(buf.toString());
056                buf.setLength(0);
057            }
058            this.outputWriter.flush();
059    
060            // Write stop bytes
061            this.outputStream.write(LLPConstants.TRAILING_BYTE_0);
062            this.outputStream.write(LLPConstants.TRAILING_BYTE_1);
063            this.outputStream.flush();
064        }
065    
066        /**
067         * Close this output stream.
068         */
069        public void close() throws IOException {
070            this.outputWriter.close();
071        }
072    }
