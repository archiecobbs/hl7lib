008    package org.dellroad.hl7.llp;
009    
010    import java.io.BufferedOutputStream;
011    import java.io.IOException;
012    import java.io.OutputStream;
013    import java.io.OutputStreamWriter;
014    import java.io.UnsupportedEncodingException;
015    
016    import org.dellroad.hl7.HL7Message;
017    import org.dellroad.hl7.HL7Segment;
018    import org.dellroad.hl7.HL7Seps;
019    import org.dellroad.hl7.HL7Writer;
020    
021    /**
022     * Writes out HL7 messages using the "lower layer protocol".
023     */
024    public class LLPOutputStream implements HL7Writer {
025    
026        private final BufferedOutputStream outputStream;
027        private final OutputStreamWriter outputWriter;
028    
029        /**
030         * Constructor.
031         *
032         * @param out underlying output stream
033         * @throws IllegalArgumentException if out is null
034         */
035        public LLPOutputStream(OutputStream out) {
036            if (out == null)
037                throw new IllegalArgumentException("out is null");
038            this.outputStream = new BufferedOutputStream(out);
039            try {
040                this.outputWriter = new OutputStreamWriter(this.outputStream, LLPConstants.CHARACTER_ENCODING);
041            } catch (UnsupportedEncodingException e) {
042                throw new RuntimeException(e);
043            }
044        }
045    
046        /**
047         * Write a message using HL7 LLP framing.
048         */
049        public void writeMessage(HL7Message message) throws IOException {
050    
051            // Write start byte
052            this.outputStream.write(LLPConstants.LEADING_BYTE);
053    
054            // Write message segments
055            HL7Seps seps = message.getMSHSegment().getHL7Seps();
056            StringBuilder buf = new StringBuilder();
057            for (HL7Segment segment : message.getSegments()) {
058                segment.append(buf, seps);
059                buf.append(HL7Message.SEGMENT_TERMINATOR);
060                this.outputWriter.write(buf.toString());
061                buf.setLength(0);
062            }
063            this.outputWriter.flush();
064    
065            // Write stop bytes
066            this.outputStream.write(LLPConstants.TRAILING_BYTE_0);
067            this.outputStream.write(LLPConstants.TRAILING_BYTE_1);
068            this.outputStream.flush();
069        }
070    
071        /**
072         * Close this output stream.
073         */
074        public void close() throws IOException {
075            this.outputWriter.close();
076        }
077    }
